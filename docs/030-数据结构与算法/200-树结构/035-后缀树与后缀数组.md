# 后缀树与后缀数组

## 概述

后缀树是紧凑的后缀Trie，用于高效处理字符串问题。后缀数组是后缀树的空间优化版本。

## 后缀树特点

1. **压缩Trie**：压缩单分支路径
2. **空间高效**：O(n)空间
3. **线性构建**：可O(n)时间构建
4. **强大功能**：支持多种字符串操作

## 后缀数组

### 数据结构

```c
typedef struct {
    char *text;
    int *sa;
    int n;
} SuffixArray;
```

### 构建后缀数组（朴素）

```c
int compareSuffix(const void *a, const void *b, void *text) {
    int i = *(int*)a;
    int j = *(int*)b;
    char *t = (char*)text;
    
    while (t[i] && t[j] && t[i] == t[j]) {
        i++;
        j++;
    }
    
    return (unsigned char)t[i] - (unsigned char)t[j];
}

SuffixArray* buildSuffixArrayNaive(char *text) {
    int n = strlen(text);
    
    SuffixArray *sa = (SuffixArray*)malloc(sizeof(SuffixArray));
    sa->text = strdup(text);
    sa->n = n;
    sa->sa = (int*)malloc(sizeof(int) * n);
    
    for (int i = 0; i < n; i++) {
        sa->sa[i] = i;
    }
    
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (strcmp(text + sa->sa[j], text + sa->sa[j + 1]) > 0) {
                int temp = sa->sa[j];
                sa->sa[j] = sa->sa[j + 1];
                sa->sa[j + 1] = temp;
            }
        }
    }
    
    return sa;
}
```

### 构建后缀数组（DC3算法简化版）

```c
int* buildSuffixArrayDC3(char *text, int n) {
    int *sa = (int*)malloc(sizeof(int) * n);
    int *rank = (int*)malloc(sizeof(int) * 2 * n);
    int *tmp = (int*)malloc(sizeof(int) * n);
    
    for (int i = 0; i < n; i++) {
        sa[i] = i;
        rank[i] = (unsigned char)text[i];
    }
    
    for (int k = 1; k < n; k *= 2) {
        auto compare = [&](int a, int b) {
            if (rank[a] != rank[b]) return rank[a] < rank[b];
            int ra = (a + k < n) ? rank[a + k] : -1;
            int rb = (b + k < n) ? rank[b + k] : -1;
            return ra < rb;
        };
        
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                int a = sa[j], b = sa[j + 1];
                int ra = rank[a], rb = rank[b];
                int rak = (a + k < n) ? rank[a + k] : -1;
                int rbk = (b + k < n) ? rank[b + k] : -1;
                
                if (ra > rb || (ra == rb && rak > rbk)) {
                    int temp = sa[j];
                    sa[j] = sa[j + 1];
                    sa[j + 1] = temp;
                }
            }
        }
        
        tmp[sa[0]] = 0;
        for (int i = 1; i < n; i++) {
            int a = sa[i - 1], b = sa[i];
            int ra = rank[a], rb = rank[b];
            int rak = (a + k < n) ? rank[a + k] : -1;
            int rbk = (b + k < n) ? rank[b + k] : -1;
            
            tmp[b] = tmp[a] + (ra != rb || rak != rbk);
        }
        
        for (int i = 0; i < n; i++) {
            rank[i] = tmp[i];
        }
        
        if (rank[sa[n - 1]] == n - 1) break;
    }
    
    free(rank);
    free(tmp);
    return sa;
}
```

### LCP数组

```c
int* buildLCP(char *text, int *sa, int n) {
    int *lcp = (int*)calloc(n, sizeof(int));
    int *rank = (int*)malloc(sizeof(int) * n);
    
    for (int i = 0; i < n; i++) {
        rank[sa[i]] = i;
    }
    
    int h = 0;
    for (int i = 0; i < n; i++) {
        if (rank[i] > 0) {
            int j = sa[rank[i] - 1];
            
            while (i + h < n && j + h < n && text[i + h] == text[j + h]) {
                h++;
            }
            
            lcp[rank[i]] = h;
            if (h > 0) h--;
        }
    }
    
    free(rank);
    return lcp;
}
```

### 子串查找

```c
int searchSubstring(SuffixArray *sa, char *pattern) {
    int m = strlen(pattern);
    int n = sa->n;
    
    int left = 0, right = n;
    
    while (left < right) {
        int mid = (left + right) / 2;
        if (strncmp(sa->text + sa->sa[mid], pattern, m) < 0) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    
    int start = left;
    right = n;
    
    while (left < right) {
        int mid = (left + right) / 2;
        if (strncmp(sa->text + sa->sa[mid], pattern, m) <= 0) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    
    return right - start;
}
```

### 最长公共子串

```c
char* longestCommonSubstring(char *s1, char *s2) {
    int n1 = strlen(s1);
    int n2 = strlen(s2);
    
    char *combined = (char*)malloc(n1 + n2 + 2);
    sprintf(combined, "%s#%s", s1, s2);
    
    int n = n1 + n2 + 1;
    int *sa = buildSuffixArrayDC3(combined, n);
    int *lcp = buildLCP(combined, sa, n);
    
    int maxLen = 0;
    int maxIndex = 0;
    
    for (int i = 1; i < n; i++) {
        int j1 = sa[i - 1];
        int j2 = sa[i];
        
        if ((j1 < n1 && j2 > n1) || (j1 > n1 && j2 < n1)) {
            if (lcp[i] > maxLen) {
                maxLen = lcp[i];
                maxIndex = sa[i];
            }
        }
    }
    
    char *result = (char*)malloc(maxLen + 1);
    strncpy(result, combined + maxIndex, maxLen);
    result[maxLen] = '\0';
    
    free(combined);
    free(sa);
    free(lcp);
    
    return result;
}
```

## C++ 实现

```cpp
#include <vector>
#include <string>
#include <algorithm>

class SuffixArray {
private:
    std::string text;
    std::vector<int> sa;
    std::vector<int> lcp;
    int n;
    
public:
    SuffixArray(const std::string& s) : text(s), n(s.length()) {
        sa.resize(n);
        lcp.resize(n);
        buildSA();
        buildLCP();
    }
    
    void buildSA() {
        for (int i = 0; i < n; i++) sa[i] = i;
        
        std::vector<int> rank(2 * n, 0);
        for (int i = 0; i < n; i++) rank[i] = text[i];
        
        for (int k = 1; k < n; k *= 2) {
            auto cmp = [&](int a, int b) {
                if (rank[a] != rank[b]) return rank[a] < rank[b];
                return rank[a + k] < rank[b + k];
            };
            
            std::sort(sa.begin(), sa.end(), cmp);
            
            std::vector<int> tmp(n);
            tmp[sa[0]] = 0;
            for (int i = 1; i < n; i++) {
                tmp[sa[i]] = tmp[sa[i - 1]] + (cmp(sa[i - 1], sa[i]) ? 1 : 0);
            }
            rank = tmp;
            
            if (rank[sa[n - 1]] == n - 1) break;
        }
    }
    
    void buildLCP() {
        std::vector<int> rank(n);
        for (int i = 0; i < n; i++) rank[sa[i]] = i;
        
        int h = 0;
        for (int i = 0; i < n; i++) {
            if (rank[i] > 0) {
                int j = sa[rank[i] - 1];
                while (i + h < n && j + h < n && text[i + h] == text[j + h]) h++;
                lcp[rank[i]] = h;
                if (h > 0) h--;
            }
        }
    }
    
    int countOccurrences(const std::string& pattern) {
        auto lower = std::lower_bound(sa.begin(), sa.end(), pattern,
            [&](int i, const std::string& p) {
                return text.compare(i, p.length(), p) < 0;
            });
        
        auto upper = std::upper_bound(sa.begin(), sa.end(), pattern,
            [&](const std::string& p, int i) {
                return text.compare(i, p.length(), p) > 0;
            });
        
        return upper - lower;
    }
};
```

## 时间复杂度

| 操作 | 后缀树 | 后缀数组 |
|------|--------|---------|
| 构建 | O(n) | O(n log n) |
| 子串查找 | O(m) | O(m log n) |
| LCP | O(1) | O(log n) |
| 空间 | O(n) | O(n) |

## 应用场景

1. **子串查找**：高效模式匹配
2. **最长重复子串**：文本分析
3. **最长公共子串**：序列比对
4. **字符串压缩**：LZ算法基础
