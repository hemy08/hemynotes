# 字典树（Trie）

## 概述

字典树（Trie）是一种树形数据结构，用于高效存储和检索字符串集合，常用于自动补全、拼写检查等场景。

## Trie结构

```
字符串集合: {app, apple, apply, banana, band}

        root
       /    \
      a      b
      |      |
      p      a
      |      |
      p*     n
     / \     |
    l   l    a
    |   |    |
    e*  y*   n
              |
              a*
              
*: 表示单词结束
```

## Trie特点

1. **公共前缀**：共享公共前缀的节点
2. **高效查找**：O(m)查找，m为字符串长度
3. **空间换时间**：可能占用较多空间
4. **前缀匹配**：适合前缀相关操作

## 基本实现

```c
#include <stdlib.h>
#include <string.h>

#define ALPHABET_SIZE 26

typedef struct TrieNode {
    struct TrieNode *children[ALPHABET_SIZE];
    int isEnd;
} TrieNode;

TrieNode* createNode() {
    TrieNode *node = (TrieNode*)malloc(sizeof(TrieNode));
    node->isEnd = 0;
    for (int i = 0; i < ALPHABET_SIZE; i++) {
        node->children[i] = NULL;
    }
    return node;
}

typedef struct {
    TrieNode *root;
} Trie;

Trie* createTrie() {
    Trie *trie = (Trie*)malloc(sizeof(Trie));
    trie->root = createNode();
    return trie;
}

void insert(Trie *trie, const char *word) {
    TrieNode *curr = trie->root;
    
    for (int i = 0; word[i]; i++) {
        int index = word[i] - 'a';
        
        if (curr->children[index] == NULL) {
            curr->children[index] = createNode();
        }
        
        curr = curr->children[index];
    }
    
    curr->isEnd = 1;
}

int search(Trie *trie, const char *word) {
    TrieNode *curr = trie->root;
    
    for (int i = 0; word[i]; i++) {
        int index = word[i] - 'a';
        
        if (curr->children[index] == NULL) {
            return 0;
        }
        
        curr = curr->children[index];
    }
    
    return curr->isEnd;
}

int startsWith(Trie *trie, const char *prefix) {
    TrieNode *curr = trie->root;
    
    for (int i = 0; prefix[i]; i++) {
        int index = prefix[i] - 'a';
        
        if (curr->children[index] == NULL) {
            return 0;
        }
        
        curr = curr->children[index];
    }
    
    return 1;
}
```

## 删除操作

```c
int isEmpty(TrieNode *node) {
    for (int i = 0; i < ALPHABET_SIZE; i++) {
        if (node->children[i] != NULL) {
            return 0;
        }
    }
    return 1;
}

TrieNode* delete(TrieNode *node, const char *word, int depth) {
    if (node == NULL) return NULL;
    
    if (word[depth] == '\0') {
        if (node->isEnd) {
            node->isEnd = 0;
        }
        
        if (isEmpty(node)) {
            free(node);
            node = NULL;
        }
        
        return node;
    }
    
    int index = word[depth] - 'a';
    node->children[index] = delete(node->children[index], word, depth + 1);
    
    if (isEmpty(node) && !node->isEnd) {
        free(node);
        node = NULL;
    }
    
    return node;
}

void deleteWord(Trie *trie, const char *word) {
    trie->root = delete(trie->root, word, 0);
}
```

## 自动补全

```c
void collectWords(TrieNode *node, char *prefix, int depth, char result[][100], int *count) {
    if (node->isEnd) {
        prefix[depth] = '\0';
        strcpy(result[*count], prefix);
        (*count)++;
    }
    
    for (int i = 0; i < ALPHABET_SIZE; i++) {
        if (node->children[i] != NULL) {
            prefix[depth] = 'a' + i;
            collectWords(node->children[i], prefix, depth + 1, result, count);
        }
    }
}

int autocomplete(Trie *trie, const char *prefix, char result[][100]) {
    TrieNode *curr = trie->root;
    
    for (int i = 0; prefix[i]; i++) {
        int index = prefix[i] - 'a';
        if (curr->children[index] == NULL) {
            return 0;
        }
        curr = curr->children[index];
    }
    
    char buffer[100];
    strcpy(buffer, prefix);
    int count = 0;
    collectWords(curr, buffer, strlen(prefix), result, &count);
    return count;
}
```

## 最长公共前缀

```c
char* longestCommonPrefix(Trie *trie, char words[][100], int n) {
    for (int i = 0; i < n; i++) {
        insert(trie, words[i]);
    }
    
    char *prefix = (char*)malloc(100);
    int len = 0;
    TrieNode *curr = trie->root;
    
    while (1) {
        int childCount = 0;
        int childIndex = -1;
        
        for (int i = 0; i < ALPHABET_SIZE; i++) {
            if (curr->children[i] != NULL) {
                childCount++;
                childIndex = i;
            }
        }
        
        if (childCount != 1 || curr->isEnd) {
            break;
        }
        
        prefix[len++] = 'a' + childIndex;
        curr = curr->children[childIndex];
    }
    
    prefix[len] = '\0';
    return prefix;
}
```

## C++ 实现

```cpp
#include <string>
#include <vector>
#include <memory>

class Trie {
private:
    struct Node {
        std::vector<std::unique_ptr<Node>> children;
        bool isEnd;
        Node() : children(26), isEnd(false) {}
    };
    
    std::unique_ptr<Node> root;
    
    void collectAll(Node* node, std::string& prefix, std::vector<std::string>& result) {
        if (node->isEnd) result.push_back(prefix);
        for (int i = 0; i < 26; i++) {
            if (node->children[i]) {
                prefix.push_back('a' + i);
                collectAll(node->children[i].get(), prefix, result);
                prefix.pop_back();
            }
        }
    }
    
public:
    Trie() : root(std::make_unique<Node>()) {}
    
    void insert(const std::string& word) {
        Node* curr = root.get();
        for (char c : word) {
            int index = c - 'a';
            if (!curr->children[index]) {
                curr->children[index] = std::make_unique<Node>();
            }
            curr = curr->children[index].get();
        }
        curr->isEnd = true;
    }
    
    bool search(const std::string& word) {
        Node* curr = root.get();
        for (char c : word) {
            int index = c - 'a';
            if (!curr->children[index]) return false;
            curr = curr->children[index].get();
        }
        return curr->isEnd;
    }
    
    bool startsWith(const std::string& prefix) {
        Node* curr = root.get();
        for (char c : prefix) {
            int index = c - 'a';
            if (!curr->children[index]) return false;
            curr = curr->children[index].get();
        }
        return true;
    }
    
    std::vector<std::string> autocomplete(const std::string& prefix) {
        Node* curr = root.get();
        for (char c : prefix) {
            int index = c - 'a';
            if (!curr->children[index]) return {};
            curr = curr->children[index].get();
        }
        
        std::vector<std::string> result;
        std::string p = prefix;
        collectAll(curr, p, result);
        return result;
    }
};
```

## 时间复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 插入 | O(m) |
| 查找 | O(m) |
| 删除 | O(m) |
| 前缀匹配 | O(m) |

m为字符串长度

## 空间复杂度

最坏 O(N × M × 26)，N为字符串数量，M为最大长度

## 应用场景

1. **自动补全**：搜索引擎、输入法
2. **拼写检查**：单词验证
3. **IP路由**：最长前缀匹配
4. **词频统计**：文本分析
5. **字符串排序**：字典序遍历
