# 最近公共祖先（LCA）

## 概述

最近公共祖先（LCA）是指树中两个节点最近的公共祖先节点。

## 暴力方法

```c
int parent[MAXN];
int depth[MAXN];

int lcaNaive(int u, int v) {
    while (depth[u] > depth[v]) u = parent[u];
    while (depth[v] > depth[u]) v = parent[v];
    while (u != v) {
        u = parent[u];
        v = parent[v];
    }
    return u;
}
```

## 倍增法

### 预处理

```c
int parent[MAXN][20];
int depth[MAXN];

void dfs(int u, int p, int d, int *adj[], int n) {
    parent[u][0] = p;
    depth[u] = d;
    
    for (int i = 0; i < n; i++) {
        if (adj[u][i] && i != p) {
            dfs(i, u, d + 1, adj, n);
        }
    }
}

void preprocess(int n, int root, int *adj[]) {
    dfs(root, -1, 0, adj, n);
    
    for (int j = 1; j < 20; j++) {
        for (int i = 0; i < n; i++) {
            if (parent[i][j - 1] != -1) {
                parent[i][j] = parent[parent[i][j - 1]][j - 1];
            } else {
                parent[i][j] = -1;
            }
        }
    }
}
```

### 查询LCA

```c
int lcaBinaryLifting(int u, int v) {
    if (depth[u] < depth[v]) {
        int temp = u;
        u = v;
        v = temp;
    }
    
    int diff = depth[u] - depth[v];
    for (int j = 0; j < 20; j++) {
        if ((diff >> j) & 1) {
            u = parent[u][j];
        }
    }
    
    if (u == v) return u;
    
    for (int j = 19; j >= 0; j--) {
        if (parent[u][j] != parent[v][j]) {
            u = parent[u][j];
            v = parent[v][j];
        }
    }
    
    return parent[u][0];
}
```

## Tarjan算法

```c
#include <string.h>

int parentSet[MAXN];
int visited[MAXN];
int ancestor[MAXN];
int lcaResult[MAXN][MAXN];

typedef struct Query {
    int v;
    int index;
    struct Query *next;
} Query;

Query *queries[MAXN];

int findSet(int x) {
    if (parentSet[x] != x) {
        parentSet[x] = findSet(parentSet[x]);
    }
    return parentSet[x];
}

void unionSet(int x, int y) {
    parentSet[findSet(x)] = findSet(y);
}

void tarjan(int u, int *adj[], int n) {
    parentSet[u] = u;
    ancestor[u] = u;
    
    for (int v = 0; v < n; v++) {
        if (adj[u][v] && !visited[v]) {
            tarjan(v, adj, n);
            unionSet(v, u);
            ancestor[findSet(u)] = u;
        }
    }
    
    visited[u] = 1;
    
    Query *q = queries[u];
    while (q != NULL) {
        if (visited[q->v]) {
            lcaResult[u][q->v] = ancestor[findSet(q->v)];
            lcaResult[q->v][u] = lcaResult[u][q->v];
        }
        q = q->next;
    }
}
```

## RMQ方法

### 欧拉序+RMQ

```c
int euler[2 * MAXN];
int depthEuler[2 * MAXN];
int firstOccur[MAXN];
int eulerCnt;
int st[2 * MAXN][20];

void dfsEuler(int u, int p, int d, int *adj[], int n) {
    firstOccur[u] = eulerCnt;
    euler[eulerCnt] = u;
    depthEuler[eulerCnt] = d;
    eulerCnt++;
    
    for (int v = 0; v < n; v++) {
        if (adj[u][v] && v != p) {
            dfsEuler(v, u, d + 1, adj, n);
            euler[eulerCnt] = u;
            depthEuler[eulerCnt] = d;
            eulerCnt++;
        }
    }
}

void buildRMQ(int n) {
    for (int i = 0; i < eulerCnt; i++) {
        st[i][0] = i;
    }
    
    for (int j = 1; (1 << j) <= eulerCnt; j++) {
        for (int i = 0; i + (1 << j) <= eulerCnt; i++) {
            int a = st[i][j - 1];
            int b = st[i + (1 << (j - 1))][j - 1];
            st[i][j] = (depthEuler[a] < depthEuler[b]) ? a : b;
        }
    }
}

int lcaRMQ(int u, int v) {
    int l = firstOccur[u];
    int r = firstOccur[v];
    if (l > r) {
        int temp = l;
        l = r;
        r = temp;
    }
    
    int k = 0;
    while ((1 << (k + 1)) <= r - l + 1) k++;
    
    int a = st[l][k];
    int b = st[r - (1 << k) + 1][k];
    
    return euler[(depthEuler[a] < depthEuler[b]) ? a : b];
}
```

## C++ 实现

```cpp
#include <vector>

class LCA {
private:
    std::vector<std::vector<int>> adj;
    std::vector<std::vector<int>> parent;
    std::vector<int> depth;
    int n, maxLog;
    
    void dfs(int u, int p, int d) {
        parent[u][0] = p;
        depth[u] = d;
        
        for (int v : adj[u]) {
            if (v != p) {
                dfs(v, u, d + 1);
            }
        }
    }
    
public:
    LCA(int n) : n(n), adj(n + 1), depth(n + 1) {
        maxLog = 0;
        while ((1 << maxLog) <= n) maxLog++;
        parent.assign(n + 1, std::vector<int>(maxLog, -1));
    }
    
    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    void build(int root) {
        dfs(root, -1, 0);
        
        for (int j = 1; j < maxLog; j++) {
            for (int i = 1; i <= n; i++) {
                if (parent[i][j - 1] != -1) {
                    parent[i][j] = parent[parent[i][j - 1]][j - 1];
                }
            }
        }
    }
    
    int lca(int u, int v) {
        if (depth[u] < depth[v]) std::swap(u, v);
        
        int diff = depth[u] - depth[v];
        for (int j = 0; j < maxLog; j++) {
            if ((diff >> j) & 1) {
                u = parent[u][j];
            }
        }
        
        if (u == v) return u;
        
        for (int j = maxLog - 1; j >= 0; j--) {
            if (parent[u][j] != parent[v][j]) {
                u = parent[u][j];
                v = parent[v][j];
            }
        }
        
        return parent[u][0];
    }
    
    int dist(int u, int v) {
        return depth[u] + depth[v] - 2 * depth[lca(u, v)];
    }
};
```

## 算法比较

| 算法 | 预处理 | 查询 | 空间 |
|------|--------|------|------|
| 暴力 | O(n) | O(n) | O(n) |
| 倍增 | O(n log n) | O(log n) | O(n log n) |
| Tarjan | O(n) | O(1) 离线 | O(n) |
| RMQ | O(n) | O(1) | O(n) |
| 树链剖分 | O(n) | O(log n) | O(n) |

## 应用场景

1. **树上距离**：dist(u,v) = depth(u) + depth(v) - 2*depth(lca)
2. **路径查询**：路径分解为三段
3. **虚树构建**：保留关键点和它们的LCA
4. **树上差分**：路径修改转单点修改
