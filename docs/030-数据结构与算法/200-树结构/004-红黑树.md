# 红黑树

## 概述

红黑树是一种自平衡二叉搜索树，通过节点颜色和旋转操作维持近似平衡。广泛应用于C++ STL的map/set、Java的TreeMap/TreeSet等。

## 红黑树性质

1. **节点颜色**：每个节点是红色或黑色
2. **根节点**：根节点是黑色
3. **叶子节点**：所有叶子节点（NIL）是黑色
4. **红色约束**：红色节点的子节点必须是黑色
5. **黑高相同**：从任一节点到其所有后代叶子的路径包含相同数量的黑色节点

## 红黑树特点

1. **近似平衡**：最长路径不超过最短路径的2倍
2. **高度上界**：h ≤ 2 * log₂(n + 1)
3. **旋转次数**：插入最多2次旋转，删除最多3次旋转
4. **效率稳定**：查找、插入、删除均为O(log n)

## 时间复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 查找 | O(log n) |
| 插入 | O(log n) |
| 删除 | O(log n) |
| 旋转 | O(1) |

## 红黑树实现

### 节点定义

```c
typedef enum { RED, BLACK } Color;

typedef struct RBNode {
    int data;
    Color color;
    struct RBNode *left;
    struct RBNode *right;
    struct RBNode *parent;
} RBNode;

RBNode* createNode(int data) {
    RBNode *node = (RBNode*)malloc(sizeof(RBNode));
    node->data = data;
    node->color = RED;
    node->left = NULL;
    node->right = NULL;
    node->parent = NULL;
    return node;
}

RBNode* NIL = NULL;

void initNIL() {
    NIL = (RBNode*)malloc(sizeof(RBNode));
    NIL->color = BLACK;
    NIL->left = NIL->right = NIL->parent = NULL;
}
```

### 左旋

```c
void leftRotate(RBNode **root, RBNode *x) {
    RBNode *y = x->right;
    x->right = y->left;
    
    if (y->left != NIL) {
        y->left->parent = x;
    }
    
    y->parent = x->parent;
    
    if (x->parent == NIL) {
        *root = y;
    } else if (x == x->parent->left) {
        x->parent->left = y;
    } else {
        x->parent->right = y;
    }
    
    y->left = x;
    x->parent = y;
}
```

### 右旋

```c
void rightRotate(RBNode **root, RBNode *y) {
    RBNode *x = y->left;
    y->left = x->right;
    
    if (x->right != NIL) {
        x->right->parent = y;
    }
    
    x->parent = y->parent;
    
    if (y->parent == NIL) {
        *root = x;
    } else if (y == y->parent->right) {
        y->parent->right = x;
    } else {
        y->parent->left = x;
    }
    
    x->right = y;
    y->parent = x;
}
```

### 插入操作

```c
void insertFixup(RBNode **root, RBNode *z) {
    while (z->parent->color == RED) {
        if (z->parent == z->parent->parent->left) {
            RBNode *y = z->parent->parent->right;
            
            if (y->color == RED) {
                z->parent->color = BLACK;
                y->color = BLACK;
                z->parent->parent->color = RED;
                z = z->parent->parent;
            } else {
                if (z == z->parent->right) {
                    z = z->parent;
                    leftRotate(root, z);
                }
                z->parent->color = BLACK;
                z->parent->parent->color = RED;
                rightRotate(root, z->parent->parent);
            }
        } else {
            RBNode *y = z->parent->parent->left;
            
            if (y->color == RED) {
                z->parent->color = BLACK;
                y->color = BLACK;
                z->parent->parent->color = RED;
                z = z->parent->parent;
            } else {
                if (z == z->parent->left) {
                    z = z->parent;
                    rightRotate(root, z);
                }
                z->parent->color = BLACK;
                z->parent->parent->color = RED;
                leftRotate(root, z->parent->parent);
            }
        }
    }
    (*root)->color = BLACK;
}

void insert(RBNode **root, int data) {
    RBNode *z = createNode(data);
    z->left = z->right = NIL;
    
    RBNode *y = NIL;
    RBNode *x = *root;
    
    while (x != NIL) {
        y = x;
        if (z->data < x->data) {
            x = x->left;
        } else {
            x = x->right;
        }
    }
    
    z->parent = y;
    
    if (y == NIL) {
        *root = z;
    } else if (z->data < y->data) {
        y->left = z;
    } else {
        y->right = z;
    }
    
    insertFixup(root, z);
}
```

### 查找操作

```c
RBNode* search(RBNode *root, int data) {
    while (root != NIL && root->data != data) {
        if (data < root->data) {
            root = root->left;
        } else {
            root = root->right;
        }
    }
    return root;
}
```

## C++ 实现（简化版）

```cpp
template<typename T>
class RedBlackTree {
private:
    enum Color { RED, BLACK };
    
    struct Node {
        T data;
        Color color;
        Node *left, *right, *parent;
        Node(T val) : data(val), color(RED), left(nullptr), right(nullptr), parent(nullptr) {}
    };
    
    Node *root;
    Node *NIL;
    
    void rotateLeft(Node *x) {
        Node *y = x->right;
        x->right = y->left;
        if (y->left != NIL) y->left->parent = x;
        y->parent = x->parent;
        if (x->parent == NIL) root = y;
        else if (x == x->parent->left) x->parent->left = y;
        else x->parent->right = y;
        y->left = x;
        x->parent = y;
    }
    
    void rotateRight(Node *y) {
        Node *x = y->left;
        y->left = x->right;
        if (x->right != NIL) x->right->parent = y;
        x->parent = y->parent;
        if (y->parent == NIL) root = x;
        else if (y == y->parent->right) y->parent->right = x;
        else y->parent->left = x;
        x->right = y;
        y->parent = x;
    }
    
    void fixInsert(Node *z) {
        while (z->parent->color == RED) {
            if (z->parent == z->parent->parent->left) {
                Node *y = z->parent->parent->right;
                if (y->color == RED) {
                    z->parent->color = BLACK;
                    y->color = BLACK;
                    z->parent->parent->color = RED;
                    z = z->parent->parent;
                } else {
                    if (z == z->parent->right) {
                        z = z->parent;
                        rotateLeft(z);
                    }
                    z->parent->color = BLACK;
                    z->parent->parent->color = RED;
                    rotateRight(z->parent->parent);
                }
            } else {
                Node *y = z->parent->parent->left;
                if (y->color == RED) {
                    z->parent->color = BLACK;
                    y->color = BLACK;
                    z->parent->parent->color = RED;
                    z = z->parent->parent;
                } else {
                    if (z == z->parent->left) {
                        z = z->parent;
                        rotateRight(z);
                    }
                    z->parent->color = BLACK;
                    z->parent->parent->color = RED;
                    rotateLeft(z->parent->parent);
                }
            }
        }
        root->color = BLACK;
    }
    
public:
    RedBlackTree() {
        NIL = new Node(T{});
        NIL->color = BLACK;
        root = NIL;
    }
    
    void insert(T data) {
        Node *z = new Node(data);
        z->left = z->right = NIL;
        Node *y = NIL, *x = root;
        while (x != NIL) {
            y = x;
            x = (data < x->data) ? x->left : x->right;
        }
        z->parent = y;
        if (y == NIL) root = z;
        else if (data < y->data) y->left = z;
        else y->right = z;
        fixInsert(z);
    }
    
    bool search(T data) {
        Node *curr = root;
        while (curr != NIL && curr->data != data)
            curr = (data < curr->data) ? curr->left : curr->right;
        return curr != NIL;
    }
};
```

## 插入修复情况

### 情况1：叔节点是红色

```
       B(祖父)              R(祖父)
      /    \               /    \
    R(父)  R(叔)   =>    B(父)  B(叔)
    |                    |
   R(z)                R(z)
   
重新着色，z上移到祖父
```

### 情况2：叔节点是黑色，z是右孩子

```
       B(祖父)         B(祖父)
      /               /
    R(父)    =>     R(z)
      \             /
      R(z)       R(父)
      
左旋父节点，转为情况3
```

### 情况3：叔节点是黑色，z是左孩子

```
       B(祖父)           B(父)
      /                /    \
    R(父)     =>     R(z)  R(祖父)
    /
  R(z)
  
右旋祖父，重新着色
```

## 红黑树验证

```c
int checkBlackHeight(RBNode *node, int *blackHeight) {
    if (node == NIL) {
        *blackHeight = 0;
        return 1;
    }
    
    int leftBH, rightBH;
    if (!checkBlackHeight(node->left, &leftBH)) return 0;
    if (!checkBlackHeight(node->right, &rightBH)) return 0;
    
    if (leftBH != rightBH) return 0;
    
    *blackHeight = leftBH + (node->color == BLACK ? 1 : 0);
    return 1;
}

int isRedBlackTree(RBNode *root) {
    if (root == NIL) return 1;
    if (root->color != BLACK) return 0;
    
    int bh;
    return checkBlackHeight(root, &bh);
}
```

## AVL vs 红黑树 vs B树

| 特性 | AVL树 | 红黑树 | B树 |
|------|-------|--------|-----|
| 平衡程度 | 严格 | 近似 | 多路 |
| 查找效率 | 最优 | 次优 | 磁盘友好 |
| 插入旋转 | 最多1次 | 最多2次 | 较少 |
| 删除旋转 | O(log n) | 最多3次 | 较少 |
| 应用场景 | 内存查找 | 通用 | 磁盘存储 |

## 应用场景

1. **关联容器**：map、set、TreeMap、TreeSet
2. **进程调度**：Linux内核CFS调度器
3. **内存管理**：malloc实现
4. **文件系统**：ext3/ext4
