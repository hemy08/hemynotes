# 布隆过滤器

## 概述

布隆过滤器是一种空间效率很高的概率型数据结构，用于判断元素是否在集合中，可能存在假阳性但不存在假阴性。

## 布隆过滤器特点

1. **空间高效**：使用位数组，空间远小于哈希表
2. **快速查询**：O(k)时间复杂度，k为哈希函数数量
3. **假阳性**：可能误判存在，但不会误判不存在
4. **不可删除**：标准布隆过滤器不支持删除

## 布隆过滤器原理

```
插入元素 "apple":
  hash1("apple") % m = 3 -> 设置bit[3] = 1
  hash2("apple") % m = 7 -> 设置bit[7] = 1
  hash3("apple") % m = 11 -> 设置bit[11] = 1

查询元素 "banana":
  hash1("banana") % m = 3 -> bit[3] = 1 ✓
  hash2("banana") % m = 5 -> bit[5] = 0 ✗
  返回"不存在"

查询元素 "orange":
  hash1("orange") % m = 3 -> bit[3] = 1 ✓
  hash2("orange") % m = 7 -> bit[7] = 1 ✓
  hash3("orange") % m = 11 -> bit[11] = 1 ✓
  返回"可能存在"（可能是假阳性）
```

## 基本实现

```c
#include <stdlib.h>
#include <string.h>
#include <math.h>

typedef struct {
    unsigned char *bits;
    int size;
    int hashCount;
} BloomFilter;

unsigned int hash1(const char *str) {
    unsigned int hash = 5381;
    int c;
    while ((c = *str++)) {
        hash = ((hash << 5) + hash) + c;
    }
    return hash;
}

unsigned int hash2(const char *str) {
    unsigned int hash = 0;
    int c;
    while ((c = *str++)) {
        hash = c + (hash << 6) + (hash << 16) - hash;
    }
    return hash;
}

unsigned int hash3(const char *str) {
    unsigned int hash = 2166136261U;
    int c;
    while ((c = *str++)) {
        hash ^= c;
        hash *= 16777619;
    }
    return hash;
}

BloomFilter* createBloomFilter(int size, int hashCount) {
    BloomFilter *bf = (BloomFilter*)malloc(sizeof(BloomFilter));
    bf->size = size;
    bf->hashCount = hashCount;
    bf->bits = (unsigned char*)calloc((size + 7) / 8, sizeof(unsigned char));
    return bf;
}

void setBit(unsigned char *bits, int index) {
    bits[index / 8] |= (1 << (index % 8));
}

int getBit(unsigned char *bits, int index) {
    return (bits[index / 8] >> (index % 8)) & 1;
}

void add(BloomFilter *bf, const char *item) {
    unsigned int h1 = hash1(item);
    unsigned int h2 = hash2(item);
    unsigned int h3 = hash3(item);
    
    setBit(bf->bits, h1 % bf->size);
    setBit(bf->bits, h2 % bf->size);
    setBit(bf->bits, h3 % bf->size);
}

int mightContain(BloomFilter *bf, const char *item) {
    unsigned int h1 = hash1(item);
    unsigned int h2 = hash2(item);
    unsigned int h3 = hash3(item);
    
    return getBit(bf->bits, h1 % bf->size) &&
           getBit(bf->bits, h2 % bf->size) &&
           getBit(bf->bits, h3 % bf->size);
}
```

## 参数计算

```c
typedef struct {
    int optimalSize;
    int optimalHashCount;
} BloomParams;

BloomParams calculateBloomParams(int expectedItems, double falsePositiveRate) {
    BloomParams params;
    
    double ln2 = log(2);
    params.optimalSize = (int)(-(expectedItems * log(falsePositiveRate)) / (ln2 * ln2));
    params.optimalHashCount = (int)((params.optimalSize / expectedItems) * ln2);
    
    if (params.optimalHashCount < 1) params.optimalHashCount = 1;
    if (params.optimalHashCount > 10) params.optimalHashCount = 10;
    
    return params;
}

BloomFilter* createOptimalBloomFilter(int expectedItems, double falsePositiveRate) {
    BloomParams params = calculateBloomParams(expectedItems, falsePositiveRate);
    return createBloomFilter(params.optimalSize, params.optimalHashCount);
}
```

## 多哈希函数实现

```c
unsigned int getHash(const char *str, int seed) {
    unsigned int hash = seed;
    int c;
    while ((c = *str++)) {
        hash = hash * 31 + c;
    }
    return hash;
}

void addMulti(BloomFilter *bf, const char *item) {
    for (int i = 0; i < bf->hashCount; i++) {
        unsigned int hash = getHash(item, i + 1);
        setBit(bf->bits, hash % bf->size);
    }
}

int mightContainMulti(BloomFilter *bf, const char *item) {
    for (int i = 0; i < bf->hashCount; i++) {
        unsigned int hash = getHash(item, i + 1);
        if (!getBit(bf->bits, hash % bf->size)) {
            return 0;
        }
    }
    return 1;
}
```

## C++ 实现

```cpp
#include <vector>
#include <string>
#include <functional>

class BloomFilter {
private:
    std::vector<bool> bits;
    int hashCount;
    
    std::vector<size_t> getHashes(const std::string& item) {
        std::vector<size_t> hashes(hashCount);
        
        size_t h1 = std::hash<std::string>{}(item);
        size_t h2 = std::hash<std::string>{}(item + "salt");
        
        for (int i = 0; i < hashCount; i++) {
            hashes[i] = h1 + i * h2;
        }
        
        return hashes;
    }
    
public:
    BloomFilter(int size, int numHashes) : bits(size, false), hashCount(numHashes) {}
    
    void add(const std::string& item) {
        auto hashes = getHashes(item);
        for (auto h : hashes) {
            bits[h % bits.size()] = true;
        }
    }
    
    bool mightContain(const std::string& item) {
        auto hashes = getHashes(item);
        for (auto h : hashes) {
            if (!bits[h % bits.size()]) return false;
        }
        return true;
    }
    
    double falsePositiveRate(int numItems) {
        return pow(1 - exp(-hashCount * numItems / bits.size()), hashCount);
    }
};
```

## 计数布隆过滤器（支持删除）

```c
typedef struct {
    int *counts;
    int size;
    int hashCount;
} CountingBloomFilter;

CountingBloomFilter* createCountingBF(int size, int hashCount) {
    CountingBloomFilter *cbf = (CountingBloomFilter*)malloc(sizeof(CountingBloomFilter));
    cbf->size = size;
    cbf->hashCount = hashCount;
    cbf->counts = (int*)calloc(size, sizeof(int));
    return cbf;
}

void addCounting(CountingBloomFilter *cbf, const char *item) {
    for (int i = 0; i < cbf->hashCount; i++) {
        unsigned int hash = getHash(item, i + 1);
        cbf->counts[hash % cbf->size]++;
    }
}

void removeCounting(CountingBloomFilter *cbf, const char *item) {
    for (int i = 0; i < cbf->hashCount; i++) {
        unsigned int hash = getHash(item, i + 1);
        if (cbf->counts[hash % cbf->size] > 0) {
            cbf->counts[hash % cbf->size]--;
        }
    }
}

int mightContainCounting(CountingBloomFilter *cbf, const char *item) {
    for (int i = 0; i < cbf->hashCount; i++) {
        unsigned int hash = getHash(item, i + 1);
        if (cbf->counts[hash % cbf->size] == 0) {
            return 0;
        }
    }
    return 1;
}
```

## 假阳性率计算

```
假阳性率 ≈ (1 - e^(-kn/m))^k

其中:
n = 插入元素数量
m = 位数组大小
k = 哈希函数数量

最优k = (m/n) * ln(2)
```

## 时间复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 插入 | O(k) |
| 查询 | O(k) |
| 空间 | O(m) |

## 应用场景

1. **缓存过滤**：减少缓存穿透
2. **网页爬虫**：URL去重
3. **垃圾邮件**：邮箱黑名单
4. **数据库**：减少磁盘IO
5. **推荐系统**：用户浏览历史
