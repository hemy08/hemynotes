# 双端队列

## 概述

双端队列（Deque）是一种两端都可以进行插入和删除操作的线性数据结构。

## 双端队列特点

1. **两端操作**：前后都可插入删除
2. **灵活性强**：可作栈、可作队列
3. **随机访问**：支持索引访问
4. **动态扩容**：容量可动态调整

## 数组实现

```c
typedef struct {
    int *data;
    int front;
    int rear;
    int capacity;
    int size;
} Deque;

Deque* createDeque(int capacity) {
    Deque *deque = (Deque*)malloc(sizeof(Deque));
    deque->data = (int*)malloc(sizeof(int) * capacity);
    deque->front = 0;
    deque->rear = capacity - 1;
    deque->capacity = capacity;
    deque->size = 0;
    return deque;
}

int isEmpty(Deque *deque) {
    return deque->size == 0;
}

int isFull(Deque *deque) {
    return deque->size == deque->capacity;
}

void pushFront(Deque *deque, int value) {
    if (isFull(deque)) return;
    
    deque->front = (deque->front - 1 + deque->capacity) % deque->capacity;
    deque->data[deque->front] = value;
    deque->size++;
}

void pushBack(Deque *deque, int value) {
    if (isFull(deque)) return;
    
    deque->rear = (deque->rear + 1) % deque->capacity;
    deque->data[deque->rear] = value;
    deque->size++;
}

int popFront(Deque *deque) {
    if (isEmpty(deque)) return -1;
    
    int value = deque->data[deque->front];
    deque->front = (deque->front + 1) % deque->capacity;
    deque->size--;
    return value;
}

int popBack(Deque *deque) {
    if (isEmpty(deque)) return -1;
    
    int value = deque->data[deque->rear];
    deque->rear = (deque->rear - 1 + deque->capacity) % deque->capacity;
    deque->size--;
    return value;
}

int getFront(Deque *deque) {
    if (isEmpty(deque)) return -1;
    return deque->data[deque->front];
}

int getBack(Deque *deque) {
    if (isEmpty(deque)) return -1;
    return deque->data[deque->rear];
}
```

## 链表实现

```c
typedef struct DequeNode {
    int data;
    struct DequeNode *prev;
    struct DequeNode *next;
} DequeNode;

typedef struct {
    DequeNode *front;
    DequeNode *rear;
    int size;
} LinkedDeque;

LinkedDeque* createLinkedDeque() {
    LinkedDeque *deque = (LinkedDeque*)malloc(sizeof(LinkedDeque));
    deque->front = NULL;
    deque->rear = NULL;
    deque->size = 0;
    return deque;
}

void pushFrontLinked(LinkedDeque *deque, int value) {
    DequeNode *node = (DequeNode*)malloc(sizeof(DequeNode));
    node->data = value;
    node->prev = NULL;
    node->next = deque->front;
    
    if (deque->front == NULL) {
        deque->rear = node;
    } else {
        deque->front->prev = node;
    }
    
    deque->front = node;
    deque->size++;
}

void pushBackLinked(LinkedDeque *deque, int value) {
    DequeNode *node = (DequeNode*)malloc(sizeof(DequeNode));
    node->data = value;
    node->next = NULL;
    node->prev = deque->rear;
    
    if (deque->rear == NULL) {
        deque->front = node;
    } else {
        deque->rear->next = node;
    }
    
    deque->rear = node;
    deque->size++;
}

int popFrontLinked(LinkedDeque *deque) {
    if (deque->front == NULL) return -1;
    
    DequeNode *node = deque->front;
    int value = node->data;
    
    deque->front = node->next;
    if (deque->front == NULL) {
        deque->rear = NULL;
    } else {
        deque->front->prev = NULL;
    }
    
    free(node);
    deque->size--;
    return value;
}

int popBackLinked(LinkedDeque *deque) {
    if (deque->rear == NULL) return -1;
    
    DequeNode *node = deque->rear;
    int value = node->data;
    
    deque->rear = node->prev;
    if (deque->rear == NULL) {
        deque->front = NULL;
    } else {
        deque->rear->next = NULL;
    }
    
    free(node);
    deque->size--;
    return value;
}
```

## C++ 实现

```cpp
template<typename T>
class Deque {
private:
    std::vector<T> data;
    int front_;
    int rear_;
    int capacity_;
    int size_;
    
    void resize() {
        std::vector<T> newData(capacity_ * 2);
        for (int i = 0; i < size_; i++) {
            newData[i] = data[(front_ + i) % capacity_];
        }
        data = std::move(newData);
        front_ = 0;
        rear_ = size_ - 1;
        capacity_ *= 2;
    }
    
public:
    Deque(int capacity = 8) : data(capacity), front_(0), rear_(capacity - 1), 
                              capacity_(capacity), size_(0) {}
    
    void pushFront(const T& value) {
        if (size_ == capacity_) resize();
        front_ = (front_ - 1 + capacity_) % capacity_;
        data[front_] = value;
        size_++;
    }
    
    void pushBack(const T& value) {
        if (size_ == capacity_) resize();
        rear_ = (rear_ + 1) % capacity_;
        data[rear_] = value;
        size_++;
    }
    
    T popFront() {
        T value = data[front_];
        front_ = (front_ + 1) % capacity_;
        size_--;
        return value;
    }
    
    T popBack() {
        T value = data[rear_];
        rear_ = (rear_ - 1 + capacity_) % capacity_;
        size_--;
        return value;
    }
    
    T& front() { return data[front_]; }
    T& back() { return data[rear_]; }
    bool empty() const { return size_ == 0; }
    int size() const { return size_; }
};
```

## 应用：滑动窗口最大值

```c
int* maxSlidingWindow(int nums[], int n, int k, int *returnSize) {
    *returnSize = n - k + 1;
    int *result = (int*)malloc(sizeof(int) * (*returnSize));
    
    Deque *deque = createDeque(n);
    
    for (int i = 0; i < n; i++) {
        while (!isEmpty(deque) && nums[getFront(deque)] <= nums[i]) {
            popFront(deque);
        }
        
        pushFront(deque, i);
        
        if (getBack(deque) <= i - k) {
            popBack(deque);
        }
        
        if (i >= k - 1) {
            result[i - k + 1] = nums[getBack(deque)];
        }
    }
    
    return result;
}
```

## 应用：表达式求值

```c
int evalRPN(char tokens[][10], int n) {
    Deque *stack = createDeque(n);
    
    for (int i = 0; i < n; i++) {
        char *token = tokens[i];
        
        if (strcmp(token, "+") == 0) {
            int b = popBack(stack);
            int a = popBack(stack);
            pushBack(stack, a + b);
        } else if (strcmp(token, "-") == 0) {
            int b = popBack(stack);
            int a = popBack(stack);
            pushBack(stack, a - b);
        } else if (strcmp(token, "*") == 0) {
            int b = popBack(stack);
            int a = popBack(stack);
            pushBack(stack, a * b);
        } else if (strcmp(token, "/") == 0) {
            int b = popBack(stack);
            int a = popBack(stack);
            pushBack(stack, a / b);
        } else {
            pushBack(stack, atoi(token));
        }
    }
    
    return getBack(stack);
}
```

## 时间复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 首端插入 | O(1) |
| 首端删除 | O(1) |
| 尾端插入 | O(1) |
| 尾端删除 | O(1) |
| 随机访问 | O(1) |

## 应用场景

1. **滑动窗口**：维护窗口最大/最小值
2. **撤销重做**：编辑器历史
3. **工作窃取**：并行调度
4. **回文检测**：双端弹出比较
