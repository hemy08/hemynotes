# 循环队列

## 概述

循环队列通过将队列首尾相连，避免假溢出，提高空间利用率。

## 循环队列特点

1. **首尾相连**：队尾可以回到队首
2. **固定容量**：预先分配固定大小
3. **避免假溢出**：空间可重复利用
4. **判满判空**：需要特殊处理

## 数组实现

```c
typedef struct {
    int *data;
    int front;
    int rear;
    int capacity;
} CircularQueue;

CircularQueue* createQueue(int capacity) {
    CircularQueue *queue = (CircularQueue*)malloc(sizeof(CircularQueue));
    queue->data = (int*)malloc(sizeof(int) * (capacity + 1));
    queue->front = 0;
    queue->rear = 0;
    queue->capacity = capacity;
    return queue;
}

int isEmpty(CircularQueue *queue) {
    return queue->front == queue->rear;
}

int isFull(CircularQueue *queue) {
    return (queue->rear + 1) % (queue->capacity + 1) == queue->front;
}

int enqueue(CircularQueue *queue, int value) {
    if (isFull(queue)) return 0;
    
    queue->data[queue->rear] = value;
    queue->rear = (queue->rear + 1) % (queue->capacity + 1);
    return 1;
}

int dequeue(CircularQueue *queue, int *value) {
    if (isEmpty(queue)) return 0;
    
    *value = queue->data[queue->front];
    queue->front = (queue->front + 1) % (queue->capacity + 1);
    return 1;
}

int front(CircularQueue *queue, int *value) {
    if (isEmpty(queue)) return 0;
    *value = queue->data[queue->front];
    return 1;
}

int size(CircularQueue *queue) {
    return (queue->rear - queue->front + queue->capacity + 1) % (queue->capacity + 1);
}
```

```c
typedef struct {

CircularQueueWithCount* createQueueCount(int capacity) {

    queue->front = 0;
    queue->rear = 0;
    queue->count = 0;
    queue->capacity = capacity;
    return queue;
}

int isEmptyCount(CircularQueueWithCount *queue) {
    return queue->count == 0;
}

int isFullCount(CircularQueueWithCount *queue) {
    return queue->count == queue->capacity;
}

void enqueueCount(CircularQueueWithCount *queue, int value) {
    if (isFullCount(queue)) return;
    
    queue->data[queue->rear] = value;
    queue->rear = (queue->rear + 1) % queue->capacity;
    queue->count++;
}

int dequeueCount(CircularQueueWithCount *queue) {
    if (isEmptyCount(queue)) return -1;
    
    int value = queue->data[queue->front];
    queue->front = (queue->front + 1) % queue->capacity;
    queue->count--;
    return value;
}
```

## 链表实现

```c
typedef struct QueueNode {
    int data;
    struct QueueNode *next;
} QueueNode;

typedef struct {
    QueueNode *front;
    QueueNode *rear;
    int size;
    int capacity;
} LinkedCircularQueue;

LinkedCircularQueue* createLinkedQueue(int capacity) {
    LinkedCircularQueue *queue = (LinkedCircularQueue*)malloc(sizeof(LinkedCircularQueue));
    queue->front = NULL;
    queue->rear = NULL;
    queue->size = 0;
    queue->capacity = capacity;
    return queue;
}

void enqueueLinked(LinkedCircularQueue *queue, int value) {
    if (queue->size >= queue->capacity) {
        QueueNode *old = queue->front;
        queue->front = queue->front->next;
        old->data = value;
        old->next = NULL;
        queue->rear->next = old;
        queue->rear = old;
        if (queue->front == NULL) queue->front = old;
    } else {
        QueueNode *node = (QueueNode*)malloc(sizeof(QueueNode));
        node->data = value;
        node->next = NULL;
        
        if (queue->rear == NULL) {
            queue->front = queue->rear = node;
        } else {
            queue->rear->next = node;
            queue->rear = node;
        }
        queue->size++;
    }
}
```

## C++ 实现

```cpp
template<typename T>
class CircularQueue {
private:
    std::vector<T> data;
    int front_;
    int rear_;
    int capacity_;
    
public:
    CircularQueue(int capacity) : data(capacity + 1), front_(0), rear_(0), capacity_(capacity) {}
    
    bool isEmpty() const { return front_ == rear_; }
    bool isFull() const { return (rear_ + 1) % (capacity_ + 1) == front_; }
    
    bool enqueue(const T& value) {
        if (isFull()) return false;
        data[rear_] = value;
        rear_ = (rear_ + 1) % (capacity_ + 1);
        return true;
    }
    
    bool dequeue(T& value) {
        if (isEmpty()) return false;
        value = data[front_];
        front_ = (front_ + 1) % (capacity_ + 1);
        return true;
    }
    
    T& front() { return data[front_]; }
    
    int size() const {
        return (rear_ - front_ + capacity_ + 1) % (capacity_ + 1);
    }
};
```

## 应用场景

### 生产者消费者模型

```c
CircularQueue *buffer;

void producer(int item) {
    while (isFull(buffer)) {
        usleep(1000);
    }
    enqueue(buffer, item);
}

int consumer() {
    int item;
    while (isEmpty(buffer)) {
        usleep(1000);
    }
    dequeue(buffer, &item);
    return item;
}
```

### 滑动窗口

```c
int slidingWindowMax(int arr[], int n, int k) {
    CircularQueue *q = createQueue(k);
    int maxSum = 0, sum = 0;
    
    for (int i = 0; i < n; i++) {
        sum += arr[i];
        enqueue(q, arr[i]);
        
        if (size(q) > k) {
            int old;
            dequeue(q, &old);
            sum -= old;
        }
        
        if (size(q) == k && sum > maxSum) {
            maxSum = sum;
        }
    }
    
    return maxSum;
}
```
