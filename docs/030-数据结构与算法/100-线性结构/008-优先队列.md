# 优先队列

## 概述

优先队列是一种特殊的队列，元素按优先级出队，而非先进先出。通常使用堆实现。

## 优先队列特点

1. **优先级出队**：最高/最低优先级元素先出
2. **动态维护**：支持动态插入删除
3. **堆实现**：通常用二叉堆实现
4. **高效操作**：插入删除O(log n)

## 数组实现（大根堆）

```c
typedef struct {
    int *data;
    int size;
    int capacity;
} PriorityQueue;

PriorityQueue* createPQ(int capacity) {
    PriorityQueue *pq = (PriorityQueue*)malloc(sizeof(PriorityQueue));
    pq->data = (int*)malloc(sizeof(int) * (capacity + 1));
    pq->size = 0;
    pq->capacity = capacity;
    return pq;
}

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

void swim(int *data, int index) {
    while (index > 1 && data[index / 2] < data[index]) {
        swap(&data[index / 2], &data[index]);
        index = index / 2;
    }
}

void sink(int *data, int size, int index) {
    while (2 * index <= size) {
        int j = 2 * index;
        if (j < size && data[j] < data[j + 1]) j++;
        if (data[index] >= data[j]) break;
        swap(&data[index], &data[j]);
        index = j;
    }
}

void push(PriorityQueue *pq, int value) {
    if (pq->size >= pq->capacity) return;
    
    pq->data[++pq->size] = value;
    swim(pq->data, pq->size);
}

int top(PriorityQueue *pq) {
    if (pq->size == 0) return -1;
    return pq->data[1];
}

int pop(PriorityQueue *pq) {
    if (pq->size == 0) return -1;
    
    int max = pq->data[1];
    pq->data[1] = pq->data[pq->size--];
    sink(pq->data, pq->size, 1);
    
    return max;
}

int isEmpty(PriorityQueue *pq) {
    return pq->size == 0;
}
```

## 小根堆实现

```c
typedef struct {
    int *data;
    int size;
    int capacity;
} MinPQ;

MinPQ* createMinPQ(int capacity) {
    MinPQ *pq = (MinPQ*)malloc(sizeof(MinPQ));
    pq->data = (int*)malloc(sizeof(int) * (capacity + 1));
    pq->size = 0;
    pq->capacity = capacity;
    return pq;
}

void swimMin(int *data, int index) {
    while (index > 1 && data[index / 2] > data[index]) {
        int temp = data[index / 2];
        data[index / 2] = data[index];
        data[index] = temp;
        index = index / 2;
    }
}

void sinkMin(int *data, int size, int index) {
    while (2 * index <= size) {
        int j = 2 * index;
        if (j < size && data[j] > data[j + 1]) j++;
        if (data[index] <= data[j]) break;
        int temp = data[index];
        data[index] = data[j];
        data[j] = temp;
        index = j;
    }
}

void pushMin(MinPQ *pq, int value) {
    pq->data[++pq->size] = value;
    swimMin(pq->data, pq->size);
}

int popMin(MinPQ *pq) {
    int min = pq->data[1];
    pq->data[1] = pq->data[pq->size--];
    sinkMin(pq->data, pq->size, 1);
    return min;
}
```

## C++ 实现

```cpp
template<typename T, typename Compare = std::less<T>>
class PriorityQueue {
private:
    std::vector<T> data;
    Compare comp;
    
    void swim(int index) {
        while (index > 0) {
            int parent = (index - 1) / 2;
            if (comp(data[parent], data[index])) {
                std::swap(data[parent], data[index]);
                index = parent;
            } else {
                break;
            }
        }
    }
    
    void sink(int index) {
        int n = data.size();
        while (2 * index + 1 < n) {
            int left = 2 * index + 1;
            int right = 2 * index + 2;
            int target = index;
            
            if (left < n && comp(data[target], data[left])) {
                target = left;
            }
            if (right < n && comp(data[target], data[right])) {
                target = right;
            }
            
            if (target == index) break;
            std::swap(data[index], data[target]);
            index = target;
        }
    }
    
public:
    PriorityQueue() = default;
    
    void push(const T& value) {
        data.push_back(value);
        swim(data.size() - 1);
    }
    
    void pop() {
        if (data.empty()) return;
        data[0] = data.back();
        data.pop_back();
        if (!data.empty()) sink(0);
    }
    
    T top() const { return data.front(); }
    bool empty() const { return data.empty(); }
    size_t size() const { return data.size(); }
};
```

## 应用：合并K个有序链表

```c
typedef struct ListNode {
    int val;
    struct ListNode *next;
} ListNode;

typedef struct {
    ListNode *node;
} HeapNode;

ListNode* mergeKLists(ListNode **lists, int k) {
    MinPQ *pq = createMinPQ(k * 100);
    
    for (int i = 0; i < k; i++) {
        if (lists[i]) {
            pushMin(pq, lists[i]->val);
        }
    }
    
    ListNode dummy = {0, NULL};
    ListNode *tail = &dummy;
    
    while (!isEmpty(pq)) {
        tail->next = (ListNode*)malloc(sizeof(ListNode));
        tail->next->val = popMin(pq);
        tail->next->next = NULL;
        tail = tail->next;
    }
    
    return dummy.next;
}
```

## 应用：Top K问题

```c
int* topKFrequent(int nums[], int n, int k, int *returnSize) {
    int freq[10000] = {0};
    
    for (int i = 0; i < n; i++) {
        freq[nums[i] + 5000]++;
    }
    
    MinPQ *pq = createMinPQ(k + 1);
    int *result = (int*)malloc(sizeof(int) * k);
    *returnSize = 0;
    
    for (int i = 0; i < 10000; i++) {
        if (freq[i] > 0) {
            pushMin(pq, freq[i]);
            if (pq->size > k) {
                popMin(pq);
            }
        }
    }
    
    while (!isEmpty(pq)) {
        result[(*returnSize)++] = popMin(pq);
    }
    
    return result;
}
```

## 应用：任务调度

```c
typedef struct {
    int id;
    int priority;
    int arrival;
    int burst;
} Task;

int compareTask(const void *a, const void *b) {
    return ((Task*)b)->priority - ((Task*)a)->priority;
}

void scheduleTasks(Task tasks[], int n) {
    PriorityQueue *pq = createPQ(n);
    
    int time = 0;
    int completed = 0;
    
    while (completed < n) {
        for (int i = 0; i < n; i++) {
            if (tasks[i].arrival <= time && tasks[i].burst > 0) {
                push(pq, tasks[i].priority);
            }
        }
        
        if (!isEmpty(pq)) {
            int priority = pop(pq);
            printf("Time %d: Execute task with priority %d\n", time, priority);
            completed++;
        }
        
        time++;
    }
}
```

## 时间复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 插入 | O(log n) |
| 删除最值 | O(log n) |
| 获取最值 | O(1) |
| 建堆 | O(n) |
