# 二分查找

## 概述

二分查找在有序数组中通过不断将搜索范围减半来查找目标元素，时间复杂度为O(log n)。

## 二分查找特点

1. **有序前提**：要求输入数组有序
2. **高效查找**：O(log n)时间复杂度
3. **跳跃访问**：非连续内存访问
4. **边界处理**：需注意整数溢出和边界条件

## 基本实现

### 递归版本

```c
int binarySearch(int arr[], int left, int right, int target) {
    if (left > right) return -1;
    
    int mid = left + (right - left) / 2;
    
    if (arr[mid] == target) return mid;
    if (arr[mid] > target) return binarySearch(arr, left, mid - 1, target);
    return binarySearch(arr, mid + 1, right, target);
}
```

### 迭代版本

```c
int binarySearch(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) return mid;
        if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    
    return -1;
}
```

## 查找边界

### 查找左边界

```c
int lowerBound(int arr[], int n, int target) {
    int left = 0, right = n;
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] < target) left = mid + 1;
        else right = mid;
    }
    
    return left;
}
```

### 查找右边界

```c
int upperBound(int arr[], int n, int target) {
    int left = 0, right = n;
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] <= target) left = mid + 1;
        else right = mid;
    }
    
    return left;
}
```

### 查找第一个等于target的位置

```c
int findFirst(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    int result = -1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            result = mid;
            right = mid - 1;
```

```c
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return result;
}
```

## 二分查找应用

### 求平方根

```c
double sqrtBinary(double x) {
    if (x < 1) return sqrtBinary(1 / x);
    
    double left = 1, right = x;
    double epsilon = 1e-10;
    
    while (right - left > epsilon) {
        double mid = left + (right - left) / 2;
        
        if (mid * mid < x) left = mid;
        else right = mid;
    }
    
    return left + (right - left) / 2;
}
```

### 搜索旋转排序数组

```c
int searchRotated(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) return mid;
        
        if (arr[left] <= arr[mid]) {
            if (target >= arr[left] && target < arr[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        } else {
            if (target > arr[mid] && target <= arr[right]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    }
    
    return -1;
}
```

### 寻找峰值

```c
int findPeakElement(int arr[], int n) {
```

### 寻找最小值（旋转数组）

```c
int findMin(int arr[], int n) {
    int left = 0, right = n - 1;
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] > arr[right]) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    
    return arr[left];
}
```

### 数组中的第K个最小元素

```c
int findKthSmallest(int arr[], int left, int right, int k) {
    if (left == right) return arr[left];
    
    int pivot = arr[right];
    int i = left;
    
    for (int j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp;
            i++;
        }
    }
    
    int temp = arr[i]; arr[i] = arr[right]; arr[right] = temp;
    
    int count = i - left + 1;
    
    if (count == k) return arr[i];
    if (count > k) return findKthSmallest(arr, left, i - 1, k);
    return findKthSmallest(arr, i + 1, right, k - count);
}
```

## 二分答案

### 吃香蕉问题

```c
int minEatingSpeed(int piles[], int n, int h) {
    int left = 1, right = 0;
    
    for (int i = 0; i < n; i++) {
        if (piles[i] > right) right = piles[i];
    }
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        
        long long total = 0;
        for (int i = 0; i < n; i++) {
            total += (piles[i] + mid - 1) / mid;
        }
        
        if (total <= h) right = mid;
        else left = mid + 1;
    }
    
    return left;
}
```

### 运输货物

```c
int shipWithinDays(int weights[], int n, int days) {
    int left = 0, right = 0;
    
    for (int i = 0; i < n; i++) {
        if (weights[i] > left) left = weights[i];
        right += weights[i];
    }
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        
        int count = 1, sum = 0;
        for (int i = 0; i < n; i++) {
            if (sum + weights[i] > mid) {
                count++;
                sum = weights[i];
            } else {
                sum += weights[i];
            }
        }
        
        if (count <= days) right = mid;
        else left = mid + 1;
    }
    
    return left;
}
```

## 常见错误

### 整数溢出

```c
int mid = (left + right) / 2;  // 可能溢出

int mid = left + (right - left) / 2;  // 正确
```

### 死循环

```c
while (left < right) {
    int mid = left + (right - left) / 2;  // 下取整
    if (arr[mid] < target) left = mid;    // 可能死循环
    else right = mid;
}

while (left < right) {
    int mid = left + (right - left + 1) / 2;  // 上取整
    if (arr[mid] > target) right = mid;
    else left = mid;
}
```
