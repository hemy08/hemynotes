# 最短路径算法

## 概述

最短路径算法用于计算图中两点之间的最短路径，主要有Dijkstra、Bellman-Ford、Floyd-Warshall等算法。

## Dijkstra算法

适用于非负权图的单源最短路径。

### 基本实现

```c
#include <limits.h>

#define V 5

int minDistance(int dist[], int visited[]) {
    int min = INT_MAX, minIndex;
    
    for (int v = 0; v < V; v++) {
        if (!visited[v] && dist[v] <= min) {
            min = dist[v];
            minIndex = v;
        }
    }
    
    return minIndex;
}

void dijkstra(int graph[V][V], int src) {
    int dist[V];
    int visited[V];
    int parent[V];
    
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        visited[i] = 0;
        parent[i] = -1;
    }
    
    dist[src] = 0;
    
    for (int count = 0; count < V - 1; count++) {
        int u = minDistance(dist, visited);
        visited[u] = 1;
        
        for (int v = 0; v < V; v++) {
            if (!visited[v] && graph[u][v] && 
                dist[u] != INT_MAX && 
                dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
                parent[v] = u;
            }
        }
    }
    
    printf("Vertex\tDistance\tPath\n");
    for (int i = 0; i < V; i++) {
        printf("%d\t%d\t\t", i, dist[i]);
        printPath(parent, i);
        printf("\n");
    }
}

void printPath(int parent[], int j) {
    if (parent[j] == -1) {
        printf("%d ", j);
        return;
    }
    printPath(parent, parent[j]);
    printf("-> %d ", j);
}
```

### 优先队列优化

```c
#include <stdlib.h>

typedef struct {
    int vertex;
    int distance;
} HeapNode;

void dijkstraOptimized(int graph[V][V], int src) {
    int dist[V];
    int parent[V];
    
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        parent[i] = -1;
    }
    
    dist[src] = 0;
    
    HeapNode heap[V * V];
    int heapSize = 0;
    heap[heapSize++] = (HeapNode){src, 0};
    
    while (heapSize > 0) {
        int minIdx = 0;
        for (int i = 1; i < heapSize; i++) {
            if (heap[i].distance < heap[minIdx].distance) {
                minIdx = i;
            }
        }
        
        HeapNode min = heap[minIdx];
        heap[minIdx] = heap[--heapSize];
        
        if (min.distance > dist[min.vertex]) continue;
        
        for (int v = 0; v < V; v++) {
            if (graph[min.vertex][v]) {
                int newDist = dist[min.vertex] + graph[min.vertex][v];
                if (newDist < dist[v]) {
                    dist[v] = newDist;
                    parent[v] = min.vertex;
                    heap[heapSize++] = (HeapNode){v, newDist};
                }
            }
        }
    }
}
```

## Bellman-Ford算法

适用于含负权边的图。

```c
void bellmanFord(int edges[][3], int E, int V, int src) {
    int dist[V];
    int parent[V];
    
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        parent[i] = -1;
    }
    dist[src] = 0;
    
    for (int i = 1; i < V; i++) {
        for (int j = 0; j < E; j++) {
            int u = edges[j][0];
            int v = edges[j][1];
            int w = edges[j][2];
            
            if (dist[u] != INT_MAX && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                parent[v] = u;
            }
        }
    }
    
    for (int j = 0; j < E; j++) {
        int u = edges[j][0];
        int v = edges[j][1];
        int w = edges[j][2];
        
        if (dist[u] != INT_MAX && dist[u] + w < dist[v]) {
            printf("Graph contains negative weight cycle\n");
            return;
        }
    }
    
    printf("Vertex\tDistance\n");
    for (int i = 0; i < V; i++) {
        printf("%d\t%d\n", i, dist[i]);
    }
}
```

## Floyd-Warshall算法

多源最短路径，动态规划实现。

```c
void floydWarshall(int graph[V][V]) {
    int dist[V][V];
    
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            dist[i][j] = graph[i][j];
        }
    }
    
    for (int k = 0; k < V; k++) {
        for (int i = 0; i < V; i++) {
            for (int j = 0; j < V; j++) {
                if (dist[i][k] != INT_MAX && dist[k][j] != INT_MAX &&
                    dist[i][k] + dist[k][j] < dist[i][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }
    
    printf("Shortest distances:\n");
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            if (dist[i][j] == INT_MAX) printf("INF ");
            else printf("%d ", dist[i][j]);
        }
        printf("\n");
    }
}
```

## SPFA算法

Bellman-Ford的队列优化版本。

```c
void spfa(int graph[V][V], int src) {
    int dist[V];
    int inQueue[V];
    int count[V] = {0};
    
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        inQueue[i] = 0;
    }
    
    int queue[V];
    int front = 0, rear = 0;
    
    dist[src] = 0;
    queue[rear++] = src;
    inQueue[src] = 1;
    
    while (front != rear) {
        int u = queue[front++];
        front %= V;
        inQueue[u] = 0;
        
        for (int v = 0; v < V; v++) {
            if (graph[u][v] != INT_MAX) {
                if (dist[u] + graph[u][v] < dist[v]) {
                    dist[v] = dist[u] + graph[u][v];
                    
                    if (!inQueue[v]) {
                        queue[rear++] = v;
                        rear %= V;
                        inQueue[v] = 1;
                        count[v]++;
                        
                        if (count[v] >= V) {
                            printf("Negative cycle detected\n");
                            return;
                        }
                    }
                }
            }
        }
    }
}
```

## 算法比较

| 算法 | 时间复杂度 | 适用场景 | 负权边 |
|------|-----------|---------|--------|
| Dijkstra | O((V+E)logV) | 非负权图 | 不支持 |
| Bellman-Ford | O(VE) | 一般图 | 支持 |
| Floyd-Warshall | O(V³) | 多源最短路径 | 支持 |
| SPFA | O(kE)平均 | 一般图 | 支持 |

## C++ 实现

```cpp
#include <vector>
#include <queue>
#include <climits>

std::vector<int> dijkstra(const std::vector<std::vector<std::pair<int,int>>>& adj, int src) {
    int n = adj.size();
    std::vector<int> dist(n, INT_MAX);
    std::priority_queue<std::pair<int,int>, std::vector<std::pair<int,int>>, 
                        std::greater<std::pair<int,int>>> pq;
    
    dist[src] = 0;
    pq.push({0, src});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if (d > dist[u]) continue;
        
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    
    return dist;
}
```
