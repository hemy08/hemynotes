# 拓扑排序

## 概述

拓扑排序是对有向无环图（DAG）顶点的一种排序，使得对于每条边(u,v)，u都出现在v之前。

## 拓扑排序特点

1. **DAG必要**：只有DAG才有拓扑排序
2. **不唯一**：一个DAG可能有多个拓扑排序
3. **线性序列**：所有顶点排成线性序列
4. **依赖关系**：适合处理依赖关系

## DFS实现

```c
#include <stdlib.h>
#include <string.h>

#define MAX_V 100

int visited[MAX_V];
int inStack[MAX_V];
int result[MAX_V];
int resultIndex;
int hasCycle;

void dfsTopo(int graph[MAX_V][MAX_V], int v, int n) {
    if (hasCycle) return;
    
    visited[v] = 1;
    inStack[v] = 1;
    
    for (int i = 0; i < n; i++) {
        if (graph[v][i]) {
            if (!visited[i]) {
                dfsTopo(graph, i, n);
            } else if (inStack[i]) {
                hasCycle = 1;
                return;
            }
        }
    }
    
    inStack[v] = 0;
    result[resultIndex++] = v;
}

int* topologicalSortDFS(int graph[MAX_V][MAX_V], int n, int *returnSize) {
    memset(visited, 0, sizeof(visited));
    memset(inStack, 0, sizeof(inStack));
    resultIndex = 0;
    hasCycle = 0;
    
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dfsTopo(graph, i, n);
        }
    }
    
    if (hasCycle) {
        *returnSize = 0;
        return NULL;
    }
    
    int *topoOrder = (int*)malloc(sizeof(int) * n);
    for (int i = 0; i < n; i++) {
        topoOrder[i] = result[n - 1 - i];
    }
    
    *returnSize = n;
    return topoOrder;
}
```

## Kahn算法（BFS）

```c
int* topologicalSortKahn(int graph[MAX_V][MAX_V], int n, int *returnSize) {
    int *inDegree = (int*)calloc(n, sizeof(int));
    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (graph[i][j]) {
                inDegree[j]++;
            }
        }
    }
    
    int *queue = (int*)malloc(sizeof(int) * n);
    int front = 0, rear = 0;
    
    for (int i = 0; i < n; i++) {
        if (inDegree[i] == 0) {
            queue[rear++] = i;
        }
    }
    
    int *result = (int*)malloc(sizeof(int) * n);
    int count = 0;
    
    while (front < rear) {
        int v = queue[front++];
        result[count++] = v;
        
        for (int i = 0; i < n; i++) {
            if (graph[v][i]) {
                inDegree[i]--;
                if (inDegree[i] == 0) {
                    queue[rear++] = i;
                }
            }
        }
    }
    
    free(inDegree);
    free(queue);
    
    if (count != n) {
        free(result);
        *returnSize = 0;
        return NULL;
    }
    
    *returnSize = n;
    return result;
}
```

## C++ 实现

```cpp
#include <vector>
#include <queue>

std::vector<int> topologicalSort(const std::vector<std::vector<int>>& adj) {
    int n = adj.size();
    std::vector<int> inDegree(n, 0);
    
    for (int u = 0; u < n; u++) {
        for (int v : adj[u]) {
            inDegree[v]++;
        }
    }
    
    std::queue<int> q;
    for (int i = 0; i < n; i++) {
        if (inDegree[i] == 0) {
            q.push(i);
        }
    }
    
    std::vector<int> result;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        result.push_back(u);
        
        for (int v : adj[u]) {
            if (--inDegree[v] == 0) {
                q.push(v);
            }
        }
    }
    
    if (result.size() != n) {
        return {};
    }
    
    return result;
}
```

## 应用：课程安排

```c
int canFinish(int numCourses, int prerequisites[][2], int prerequisitesSize) {
    int **graph = (int**)malloc(sizeof(int*) * numCourses);
    for (int i = 0; i < numCourses; i++) {
        graph[i] = (int*)calloc(numCourses, sizeof(int));
    }
    
    for (int i = 0; i < prerequisitesSize; i++) {
        int course = prerequisites[i][0];
        int prereq = prerequisites[i][1];
        graph[prereq][course] = 1;
    }
    
    int returnSize;
    int *order = topologicalSortKahn(graph, numCourses, &returnSize);
    
    for (int i = 0; i < numCourses; i++) {
        free(graph[i]);
    }
    free(graph);
    
    if (order == NULL) {
        return 0;
    }
    
    free(order);
    return 1;
}

int* findOrder(int numCourses, int prerequisites[][2], int prerequisitesSize, int *returnSize) {
    int **graph = (int**)malloc(sizeof(int*) * numCourses);
    for (int i = 0; i < numCourses; i++) {
        graph[i] = (int*)calloc(numCourses, sizeof(int));
    }
    
    for (int i = 0; i < prerequisitesSize; i++) {
        graph[prerequisites[i][1]][prerequisites[i][0]] = 1;
    }
    
    int *order = topologicalSortKahn(graph, numCourses, returnSize);
    
    for (int i = 0; i < numCourses; i++) {
        free(graph[i]);
    }
    free(graph);
    
    return order;
}
```

## 应用：关键路径

```c
void criticalPath(int graph[MAX_V][MAX_V], int n, int weights[MAX_V][MAX_V]) {
    int *topoOrder = (int*)malloc(sizeof(int) * n);
    int returnSize;
    
    int *result = topologicalSortKahn(graph, n, &returnSize);
    if (result == NULL) {
        printf("Graph has cycle\n");
        return;
    }
    
    int *earliest = (int*)calloc(n, sizeof(int));
    int *latest = (int*)malloc(sizeof(int) * n);
    
    for (int i = 0; i < n; i++) {
        int v = result[i];
        for (int j = 0; j < n; j++) {
            if (graph[j][v]) {
                int finish = earliest[j] + weights[j][v];
                if (finish > earliest[v]) {
                    earliest[v] = finish;
                }
            }
        }
    }
    
    for (int i = 0; i < n; i++) {
        latest[i] = earliest[result[n - 1]];
    }
    
    for (int i = n - 1; i >= 0; i--) {
        int v = result[i];
        for (int j = 0; j < n; j++) {
            if (graph[v][j]) {
                int start = latest[j] - weights[v][j];
                if (start < latest[v]) {
                    latest[v] = start;
                }
            }
        }
    }
    
    printf("Critical activities:\n");
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (graph[i][j]) {
                int e = earliest[i];
                int l = latest[j] - weights[i][j];
                if (e == l) {
                    printf("%d -> %d\n", i, j);
                }
            }
        }
    }
    
    free(topoOrder);
    free(earliest);
    free(latest);
    free(result);
}
```
