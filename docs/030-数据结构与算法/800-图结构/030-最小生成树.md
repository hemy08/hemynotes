# 最小生成树

## 概述

最小生成树（MST）是连通图中边权值之和最小的生成树。主要算法有Prim和Kruskal。

## Prim算法

从任意顶点开始，每次选择连接已选顶点和未选顶点的最小边。

### 基本实现

```c
#include <limits.h>

#define V 5

int minKey(int key[], int mstSet[]) {
    int min = INT_MAX, minIndex;
    
    for (int v = 0; v < V; v++) {
        if (!mstSet[v] && key[v] < min) {
            min = key[v];
            minIndex = v;
        }
    }
    
    return minIndex;
}

void primMST(int graph[V][V]) {
    int parent[V];
    int key[V];
    int mstSet[V];
    
    for (int i = 0; i < V; i++) {
        key[i] = INT_MAX;
        mstSet[i] = 0;
    }
    
    key[0] = 0;
    parent[0] = -1;
    
    for (int count = 0; count < V - 1; count++) {
        int u = minKey(key, mstSet);
        mstSet[u] = 1;
        
        for (int v = 0; v < V; v++) {
            if (graph[u][v] && !mstSet[v] && graph[u][v] < key[v]) {
                parent[v] = u;
                key[v] = graph[u][v];
            }
        }
    }
    
    printf("Edge\tWeight\n");
    for (int i = 1; i < V; i++) {
        printf("%d - %d\t%d\n", parent[i], i, graph[i][parent[i]]);
    }
}
```

### 优先队列优化

```c
typedef struct {
    int vertex;
    int key;
} HeapNode;

void primOptimized(int graph[V][V]) {
    int parent[V];
    int key[V];
    int mstSet[V];
    
    for (int i = 0; i < V; i++) {
        key[i] = INT_MAX;
        mstSet[i] = 0;
        parent[i] = -1;
    }
    
    key[0] = 0;
    
    HeapNode heap[V];
    int heapSize = V;
    for (int i = 0; i < V; i++) {
        heap[i] = (HeapNode){i, key[i]};
    }
    
    while (heapSize > 0) {
        int minIdx = 0;
        for (int i = 1; i < heapSize; i++) {
            if (heap[i].key < heap[minIdx].key) {
                minIdx = i;
            }
        }
        
        HeapNode min = heap[minIdx];
        heap[minIdx] = heap[--heapSize];
        
        int u = min.vertex;
        mstSet[u] = 1;
        
        for (int v = 0; v < V; v++) {
            if (graph[u][v] && !mstSet[v] && graph[u][v] < key[v]) {
                key[v] = graph[u][v];
                parent[v] = u;
                
                for (int i = 0; i < heapSize; i++) {
                    if (heap[i].vertex == v) {
                        heap[i].key = key[v];
                        break;
                    }
                }
            }
        }
    }
}
```

## Kruskal算法

按边权值从小到大选择边，保证不形成环。

### 基本实现

```c
typedef struct {
    int src, dest, weight;
} Edge;

int parent[V];

int find(int i) {
    while (parent[i] != i) {
        i = parent[i];
    }
    return i;
}

void unionSets(int i, int j) {
    int a = find(i);
    int b = find(j);
    parent[a] = b;
}

int compareEdges(const void *a, const void *b) {
    return ((Edge*)a)->weight - ((Edge*)b)->weight;
}

void kruskalMST(Edge edges[], int E) {
    qsort(edges, E, sizeof(Edge), compareEdges);
    
    for (int i = 0; i < V; i++) {
        parent[i] = i;
    }
    
    Edge result[V - 1];
    int e = 0;
    int i = 0;
    
    while (e < V - 1 && i < E) {
        Edge next = edges[i++];
        
        int x = find(next.src);
        int y = find(next.dest);
        
        if (x != y) {
            result[e++] = next;
            unionSets(x, y);
        }
    }
    
    printf("Edge\tWeight\n");
    for (i = 0; i < e; i++) {
        printf("%d - %d\t%d\n", result[i].src, result[i].dest, result[i].weight);
    }
}
```

### 路径压缩优化

```c
int findCompressed(int i) {
    if (parent[i] != i) {
        parent[i] = findCompressed(parent[i]);
    }
    return parent[i];
}

void unionByRank(int i, int j, int rank[]) {
    int rootI = findCompressed(i);
    int rootJ = findCompressed(j);
    
    if (rank[rootI] < rank[rootJ]) {
        parent[rootI] = rootJ;
    } else if (rank[rootI] > rank[rootJ]) {
        parent[rootJ] = rootI;
    } else {
        parent[rootJ] = rootI;
        rank[rootI]++;
    }
}
```

## C++ 实现

```cpp
#include <vector>
#include <queue>
#include <algorithm>

class MST {
private:
    int V;
    std::vector<std::vector<std::pair<int, int>>> adj;
    
public:
    MST(int vertices) : V(vertices), adj(vertices) {}
    
    void addEdge(int u, int v, int w) {
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    
    std::vector<std::tuple<int, int, int>> prim() {
        std::vector<int> key(V, INT_MAX);
        std::vector<int> parent(V, -1);
        std::vector<bool> mstSet(V, false);
        
        std::priority_queue<std::pair<int, int>, 
                           std::vector<std::pair<int, int>>,
                           std::greater<std::pair<int, int>>> pq;
        
        key[0] = 0;
        pq.push({0, 0});
        
        while (!pq.empty()) {
            int u = pq.top().second;
            pq.pop();
            
            if (mstSet[u]) continue;
            mstSet[u] = true;
            
            for (auto [v, w] : adj[u]) {
                if (!mstSet[v] && w < key[v]) {
                    key[v] = w;
                    parent[v] = u;
                    pq.push({key[v], v});
                }
            }
        }
        
        std::vector<std::tuple<int, int, int>> result;
        for (int i = 1; i < V; i++) {
            result.push_back({parent[i], i, key[i]});
        }
        
        return result;
    }
};
```

## 算法比较

| 特性 | Prim | Kruskal |
|------|------|---------|
| 时间复杂度 | O(E log V) | O(E log E) |
| 数据结构 | 优先队列 | 并查集 |
| 适用图类型 | 稠密图 | 稀疏图 |
| 实现难度 | 中等 | 简单 |

## Boruvka算法

适用于并行计算。

```c
void boruvkaMST(Edge edges[], int E) {
    int parent[V];
    int rank[V];
    
    for (int i = 0; i < V; i++) {
        parent[i] = i;
        rank[i] = 0;
    }
    
    int numTrees = V;
    
    while (numTrees > 1) {
        int cheapest[V];
        for (int i = 0; i < V; i++) {
            cheapest[i] = -1;
        }
        
        for (int i = 0; i < E; i++) {
            int set1 = findCompressed(edges[i].src);
            int set2 = findCompressed(edges[i].dest);
            
            if (set1 != set2) {
                if (cheapest[set1] == -1 || 
                    edges[cheapest[set1]].weight > edges[i].weight) {
                    cheapest[set1] = i;
                }
                if (cheapest[set2] == -1 || 
                    edges[cheapest[set2]].weight > edges[i].weight) {
                    cheapest[set2] = i;
                }
            }
        }
        
        for (int i = 0; i < V; i++) {
            if (cheapest[i] != -1) {
                int set1 = findCompressed(edges[cheapest[i]].src);
                int set2 = findCompressed(edges[cheapest[i]].dest);
                
                if (set1 != set2) {
                    unionByRank(set1, set2, rank);
                    printf("%d - %d: %d\n", 
                           edges[cheapest[i]].src,
                           edges[cheapest[i]].dest,
                           edges[cheapest[i]].weight);
                    numTrees--;
                }
            }
        }
    }
}
```
