# 二分图

## 概述

二分图是图论中的重要概念，顶点可分为两个不相交集合，所有边连接两个集合中的顶点。

## 二分图性质

1. **顶点划分**：可分为两个独立集
2. **边连接**：所有边连接两个集合
3. **无奇环**：不包含奇数长度的环
4. **等价条件**：可二着色

## 染色法判定

```c
#define MAX_V 100

int colors[MAX_V];

int isBipartiteDFS(int graph[MAX_V][MAX_V], int v, int color, int n) {
    colors[v] = color;
    
    for (int i = 0; i < n; i++) {
        if (graph[v][i]) {
            if (colors[i] == -1) {
                if (!isBipartiteDFS(graph, i, 1 - color, n)) {
                    return 0;
                }
            } else if (colors[i] == color) {
                return 0;
            }
        }
    }
    
    return 1;
}

int isBipartite(int graph[MAX_V][MAX_V], int n) {
    for (int i = 0; i < n; i++) {
        colors[i] = -1;
    }
    
    for (int i = 0; i < n; i++) {
        if (colors[i] == -1) {
            if (!isBipartiteDFS(graph, i, 0, n)) {
                return 0;
            }
        }
    }
    
    return 1;
}
```

## BFS判定

```c
int isBipartiteBFS(int graph[MAX_V][MAX_V], int n) {
    int color[MAX_V];
    for (int i = 0; i < n; i++) color[i] = -1;
    
    for (int start = 0; start < n; start++) {
        if (color[start] != -1) continue;
        
        int queue[MAX_V];
        int front = 0, rear = 0;
        
        queue[rear++] = start;
        color[start] = 0;
        
        while (front < rear) {
            int v = queue[front++];
            
            for (int i = 0; i < n; i++) {
                if (graph[v][i]) {
                    if (color[i] == -1) {
                        color[i] = 1 - color[v];
                        queue[rear++] = i;
                    } else if (color[i] == color[v]) {
                        return 0;
                    }
                }
            }
        }
    }
    
    return 1;
}
```

## 二分图最大匹配（匈牙利算法）

```c
int match[MAX_V];
int visited[MAX_V];

int dfs(int graph[MAX_V][MAX_V], int u, int n, int m) {
    for (int v = 0; v < m; v++) {
        if (graph[u][v] && !visited[v]) {
            visited[v] = 1;
            
            if (match[v] == -1 || dfs(graph, match[v], n, m)) {
                match[v] = u;
                return 1;
            }
        }
    }
    return 0;
}

int maxMatching(int graph[MAX_V][MAX_V], int n, int m) {
    for (int i = 0; i < m; i++) match[i] = -1;
    
    int result = 0;
    
    for (int u = 0; u < n; u++) {
        for (int i = 0; i < m; i++) visited[i] = 0;
        
        if (dfs(graph, u, n, m)) {
            result++;
        }
    }
    
    return result;
}
```

int canAssign(int graph[MAX_V][MAX_V], int n, int m, int assign[]) {
    for (int i = 0; i < m; i++) match[i] = -1;
    
    for (int u = 0; u < n; u++) {
        for (int i = 0; i < m; i++) visited[i] = 0;
        
        if (!dfs(graph, u, n, m)) {
            return 0;

}

## 最大权匹配（KM算法）

```c
#define INF 1000000000

int lx[MAX_V], ly[MAX_V];
int slack[MAX_V];

int dfsKM(int graph[MAX_V][MAX_V], int u, int n) {
    visited[u] = 1;
    
    for (int v = 0; v < n; v++) {
        if (!visited[v]) {
            int t = lx[u] + ly[v] - graph[u][v];
            
            if (t == 0) {
                visited[v] = 1;
                if (match[v] == -1 || dfsKM(graph, match[v], n)) {
                    match[v] = u;
                    return 1;
                }
            } else if (slack[v] > t) {
                slack[v] = t;
            }
        }
    }
    return 0;
}

int maxWeightMatching(int graph[MAX_V][MAX_V], int n) {
    for (int i = 0; i < n; i++) {
        lx[i] = -INF;
        ly[i] = 0;
        match[i] = -1;
        
        for (int j = 0; j < n; j++) {
            if (graph[i][j] > lx[i]) {
                lx[i] = graph[i][j];
            }
        }
    }
    
    for (int u = 0; u < n; u++) {
        for (int i = 0; i < n; i++) slack[i] = INF;
        
        while (1) {
            for (int i = 0; i < n; i++) visited[i] = 0;
            
            if (dfsKM(graph, u, n)) break;
            
            int d = INF;
            for (int i = 0; i < n; i++) {
                if (!visited[i] && slack[i] < d) {
                    d = slack[i];
                }
            }
            
            for (int i = 0; i < n; i++) {
                if (visited[i]) lx[i] -= d;
                if (!visited[i]) ly[i] += d;
            }
        }
    }
    
    int result = 0;
    for (int i = 0; i < n; i++) {
        if (match[i] != -1) {
            result += graph[match[i]][i];
        }
    }
    
    return result;
}
```

## C++ 实现

```cpp
#include <vector>

class BipartiteMatching {
private:
    std::vector<std::vector<int>> adj;
    std::vector<int> match;
    std::vector<bool> visited;
    int n, m;
    
    bool dfs(int u) {
        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                if (match[v] == -1 || dfs(match[v])) {
                    match[v] = u;
                    return true;
                }
            }
        }
        return false;
    }
    
public:
    BipartiteMatching(int leftSize, int rightSize) : n(leftSize), m(rightSize), adj(leftSize) {}
    
    void addEdge(int u, int v) {
        adj[u].push_back(v);
    }
    
    int maxMatching() {
        match.assign(m, -1);
        int result = 0;
        
        for (int u = 0; u < n; u++) {
            visited.assign(m, false);
            if (dfs(u)) result++;
        }
        
        return result;
    }
    
    std::vector<std::pair<int, int>> getMatching() {
        std::vector<std::pair<int, int>> result;
        for (int v = 0; v < m; v++) {
            if (match[v] != -1) {
                result.push_back({match[v], v});
            }
        }
        return result;
    }
};
```
