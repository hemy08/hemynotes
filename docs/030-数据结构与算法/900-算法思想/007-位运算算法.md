# 位运算算法

## 概述

位运算直接操作二进制位，效率极高，常用于优化算法、状态压缩、权限控制等场景。

## 基本位运算

| 运算 | 符号 | 说明 |
|------|------|------|
| 与 | & | 两位都为1则结果为1 |
| 或 | \| | 有一位为1则结果为1 |
| 异或 | ^ | 两位不同则结果为1 |
| 取反 | ~ | 0变1，1变0 |
| 左移 | << | 各位左移，低位补0 |
| 右移 | >> | 各位右移，高位补符号位 |

## 常用位运算技巧

### 判断奇偶

```c
int isOdd(int n) {
    return n & 1;
}

int isEven(int n) {
    return !(n & 1);
}
```

### 交换两数

```c
void swap(int *a, int *b) {
    *a ^= *b;
    *b ^= *a;
    *a ^= *b;
}
```

### 取第k位

```c
int getBit(int n, int k) {
    return (n >> k) & 1;
}
```

### 设置第k位为1

```c
int setBit(int n, int k) {
    return n | (1 << k);
}
```

### 清除第k位

```c
int clearBit(int n, int k) {
    return n & ~(1 << k);
}
```

### 翻转第k位

```c
int toggleBit(int n, int k) {
    return n ^ (1 << k);
}
```

### 最低位1

```c
int lowbit(int n) {
    return n & (-n);
}
```

### 判断是否为2的幂

```c
int isPowerOfTwo(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}
```

### 统计1的个数

```c
int countBits(int n) {
    int count = 0;
    while (n) {
        count += n & 1;
        n >>= 1;
    }
    return count;
}

int countBitsOptimized(int n) {
    int count = 0;
    while (n) {
        n &= n - 1;
        count++;
    }
    return count;
}
```

## 位运算应用

### 只出现一次的数字

```c
int singleNumber(int nums[], int n) {
    int result = 0;
    for (int i = 0; i < n; i++) {
        result ^= nums[i];
    }
    return result;
}
```

### 只出现一次的数字（其他出现三次）

```c
int singleNumberII(int nums[], int n) {
    int ones = 0, twos = 0;
    
    for (int i = 0; i < n; i++) {
        ones = (ones ^ nums[i]) & ~twos;
        twos = (twos ^ nums[i]) & ~ones;
    }
    
    return ones;
}
```

### 两个只出现一次的数字

```c
void singleNumberIII(int nums[], int n, int *a, int *b) {
    int xor = 0;
    for (int i = 0; i < n; i++) {
        xor ^= nums[i];
    }
    
    int diff = xor & (-xor);
    
    *a = 0;
    *b = 0;
    for (int i = 0; i < n; i++) {
        if (nums[i] & diff) {
            *a ^= nums[i];
        } else {
            *b ^= nums[i];
        }
    }
}
```

### 颠倒二进制位

```c
unsigned int reverseBits(unsigned int n) {
    unsigned int result = 0;
    for (int i = 0; i < 32; i++) {
        result = (result << 1) | (n & 1);
        n >>= 1;
    }
    return result;
}

unsigned int reverseBitsOptimized(unsigned int n) {
    n = (n >> 16) | (n << 16);
    n = ((n & 0xff00ff00) >> 8) | ((n & 0x00ff00ff) << 8);
    n = ((n & 0xf0f0f0f0) >> 4) | ((n & 0x0f0f0f0f) << 4);
    n = ((n & 0xcccccccc) >> 2) | ((n & 0x33333333) << 2);
    n = ((n & 0xaaaaaaaa) >> 1) | ((n & 0x55555555) << 1);
    return n;
}
```

### 求两数平均值（避免溢出）

```c
int average(int a, int b) {
    return (a & b) + ((a ^ b) >> 1);
}
```

### 判断符号是否相同

```c
int sameSign(int a, int b) {
    return (a ^ b) >= 0;
}
```

### 取绝对值

```c
int absBit(int n) {
    int mask = n >> 31;
    return (n ^ mask) - mask;
}
```

### 最大最小值

```c
int maxBit(int a, int b) {
    return a - ((a - b) & ((a - b) >> 31));
}

int minBit(int a, int b) {
    return b + ((a - b) & ((a - b) >> 31));
}
```

## 状态压缩DP

### 子集枚举

```c
void enumerateSubsets(int n) {
    for (int mask = 0; mask < (1 << n); mask++) {
        printf("Subset %d: ", mask);
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) {
                printf("%d ", i);
            }
        }
        printf("\n");
    }
}
```

### 枚举子集的子集

```c
void enumerateSubSubsets(int mask) {
    for (int sub = mask; sub; sub = (sub - 1) & mask) {
        printf("Sub-subset: %d\n", sub);
    }
}
```

### 枚举大小为k的子集

```c
int nextCombination(int comb, int n) {
    int x = comb & -comb;
    int y = comb + x;
    return ((comb & ~y) / x >> 1) | y;
}

void enumerateKCombinations(int n, int k) {
    int comb = (1 << k) - 1;
    while (comb < (1 << n)) {
        printf("Combination: ");
        for (int i = 0; i < n; i++) {
            if (comb & (1 << i)) printf("%d ", i);
        }
        printf("\n");
        comb = nextCombination(comb, n);
    }
}
```

## 权限控制

```c
#define PERMISSION_READ    1  // 0001
#define PERMISSION_WRITE   2  // 0010
#define PERMISSION_EXECUTE 4  // 0100
#define PERMISSION_DELETE  8  // 1000

int hasPermission(int userPerms, int perm) {
    return (userPerms & perm) != 0;
}

int grantPermission(int userPerms, int perm) {
    return userPerms | perm;
}

int revokePermission(int userPerms, int perm) {
    return userPerms & ~perm;
}
```
