# 字符串匹配算法

## 概述

字符串匹配是在文本串中查找模式串出现位置的问题，常用算法有暴力匹配、KMP、BM、Sunday等。

## 暴力匹配

```c
int bruteForce(const char *text, const char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    
    for (int i = 0; i <= n - m; i++) {
        int j;
        for (j = 0; j < m; j++) {
            if (text[i + j] != pattern[j]) {
                break;
            }
        }
        if (j == m) {
            return i;
        }
    }
    
    return -1;
}
```

## KMP算法

### 基本实现

```c
void computeLPS(const char *pattern, int *lps) {
    int m = strlen(pattern);
    int len = 0;
    lps[0] = 0;
    
    int i = 1;
    while (i < m) {
        if (pattern[i] == pattern[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len != 0) {
                len = lps[len - 1];
            } else {
                lps[i] = 0;
                i++;
            }
        }
    }
}

int kmpSearch(const char *text, const char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    
    int *lps = (int*)malloc(sizeof(int) * m);
    computeLPS(pattern, lps);
    
    int i = 0, j = 0;
    while (i < n) {
        if (pattern[j] == text[i]) {
            i++;
            j++;
        }
        
        if (j == m) {
            free(lps);
            return i - j;
        } else if (i < n && pattern[j] != text[i]) {
            if (j != 0) {
                j = lps[j - 1];
            } else {
                i++;
            }
        }
    }
    
    free(lps);
    return -1;
}
```

### 查找所有匹配位置

```c
int* kmpFindAll(const char *text, const char *pattern, int *count) {
    int n = strlen(text);
    int m = strlen(pattern);
    
    int *lps = (int*)malloc(sizeof(int) * m);
    computeLPS(pattern, lps);
    
    int *result = (int*)malloc(sizeof(int) * n);
    *count = 0;
    
    int i = 0, j = 0;
    while (i < n) {
        if (pattern[j] == text[i]) {
            i++;
            j++;
        }
        
        if (j == m) {
            result[(*count)++] = i - j;
            j = lps[j - 1];
        } else if (i < n && pattern[j] != text[i]) {
            if (j != 0) {
                j = lps[j - 1];
            } else {
                i++;
            }
        }
    }
    
    free(lps);
    return result;
}
```

## BM算法

```c
void badCharHeuristic(const char *pattern, int m, int badChar[256]) {
    for (int i = 0; i < 256; i++) {
        badChar[i] = -1;
    }
    
    for (int i = 0; i < m; i++) {
        badChar[(unsigned char)pattern[i]] = i;
    }
}

int bmSearch(const char *text, const char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    
    int badChar[256];
    badCharHeuristic(pattern, m, badChar);
    
    int s = 0;
    while (s <= n - m) {
        int j = m - 1;
        
        while (j >= 0 && pattern[j] == text[s + j]) {
            j--;
        }
        
        if (j < 0) {
            return s;
        } else {
            int bcShift = j - badChar[(unsigned char)text[s + j]];
            s += (bcShift > 1) ? bcShift : 1;
        }
    }
    
    return -1;
}
```

## Sunday算法

```c
void sundayHeuristic(const char *pattern, int m, int shift[256]) {
    for (int i = 0; i < 256; i++) {
        shift[i] = m + 1;
    }
    
    for (int i = 0; i < m; i++) {
        shift[(unsigned char)pattern[i]] = m - i;
    }
}

int sundaySearch(const char *text, const char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    
    int shift[256];
    sundayHeuristic(pattern, m, shift);
    
    int i = 0;
    while (i <= n - m) {
        int j;
        for (j = 0; j < m; j++) {
            if (text[i + j] != pattern[j]) {
                break;
            }
        }
        
        if (j == m) {
            return i;
        }
        
        if (i + m >= n) break;
        
        i += shift[(unsigned char)text[i + m]];
    }
    
    return -1;
}
```

## Rabin-Karp算法

```c
int rabinKarp(const char *text, const char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    int d = 256;
    int q = 101;
    
    int h = 1;
    for (int i = 0; i < m - 1; i++) {
        h = (h * d) % q;
    }
    
    int p = 0;
    int t = 0;
    
    for (int i = 0; i < m; i++) {
        p = (d * p + pattern[i]) % q;
        t = (d * t + text[i]) % q;
    }
    
    for (int i = 0; i <= n - m; i++) {
        if (p == t) {
            int j;
            for (j = 0; j < m; j++) {
                if (text[i + j] != pattern[j]) {
                    break;
                }
            }
            if (j == m) {
                return i;
            }
        }
        
        if (i < n - m) {
            t = (d * (t - text[i] * h) + text[i + m]) % q;
            if (t < 0) t += q;
        }
    }
    
    return -1;
}
```

## C++ 实现

```cpp
#include <string>
#include <vector>

class KMP {
private:
    std::string pattern;
    std::vector<int> lps;
    
    void computeLPS() {
        int m = pattern.length();
        lps.resize(m);
        int len = 0;
        
        for (int i = 1; i < m; i++) {
            while (len > 0 && pattern[i] != pattern[len]) {
                len = lps[len - 1];
            }
            if (pattern[i] == pattern[len]) {
                len++;
            }
            lps[i] = len;
        }
    }
    
public:
    KMP(const std::string& p) : pattern(p) {
        computeLPS();
    }
    
    int search(const std::string& text) {
        int n = text.length();
        int m = pattern.length();
        
        for (int i = 0, j = 0; i < n; i++) {
            while (j > 0 && text[i] != pattern[j]) {
                j = lps[j - 1];
            }
            if (text[i] == pattern[j]) {
                j++;
            }
            if (j == m) {
                return i - m + 1;
            }
        }
        return -1;
    }
    
    std::vector<int> findAll(const std::string& text) {
        std::vector<int> result;
        int n = text.length();
        int m = pattern.length();
        
        for (int i = 0, j = 0; i < n; i++) {
            while (j > 0 && text[i] != pattern[j]) {
                j = lps[j - 1];
            }
            if (text[i] == pattern[j]) {
                j++;
            }
            if (j == m) {
                result.push_back(i - m + 1);
                j = lps[j - 1];
            }
        }
        return result;
    }
};
```

## 算法比较

| 算法 | 预处理 | 匹配时间 | 特点 |
|------|--------|---------|------|
| 暴力 | O(1) | O(nm) | 简单 |
| KMP | O(m) | O(n) | 无回溯 |
| BM | O(m+σ) | O(n/m)最好 | 实践最快 |
| Sunday | O(m+σ) | O(n) | 简单高效 |
| Rabin-Karp | O(m) | O(n+m) | 哈希匹配 |

## 应用场景

1. **文本编辑**：查找替换

3. **DNA序列**：生物信息学
4. **入侵检测**：模式匹配
