# 动态规划

## 概述

动态规划通过将问题分解为重叠子问题，存储子问题的解避免重复计算，适用于具有最优子结构性质的问题。

## 动态规划要素

1. **最优子结构**：问题的最优解包含子问题的最优解
2. **重叠子问题**：子问题会被多次计算
3. **无后效性**：当前状态一旦确定，之前的状态不影响之后的决策

## 解题步骤

1. 定义状态
2. 建立状态转移方程
3. 确定初始条件
4. 确定计算顺序
5. 求解最终结果

## 经典问题

### 斐波那契数列

```c
long long fib[1000];

long long fibonacciDP(int n) {
    if (n <= 1) return n;
    
    fib[0] = 0;
    fib[1] = 1;
    
    for (int i = 2; i <= n; i++) {
        fib[i] = fib[i - 1] + fib[i - 2];
    }
    
    return fib[n];
}
```

### 爬楼梯

```c
int climbStairs(int n) {
    if (n <= 2) return n;
    
    int dp[n + 1];
    dp[1] = 1;
    dp[2] = 2;
    
    for (int i = 3; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

int climbStairsOptimized(int n) {
    if (n <= 2) return n;
    
    int a = 1, b = 2;
    for (int i = 3; i <= n; i++) {
        int temp = a + b;
        a = b;
        b = temp;
    }
    return b;
}
```

### 0-1背包问题

```c
int knapsack(int weights[], int values[], int n, int W) {
    int dp[n + 1][W + 1];
    
    for (int i = 0; i <= n; i++) {
        for (int w = 0; w <= W; w++) {
            if (i == 0 || w == 0) {
                dp[i][w] = 0;
            } else if (weights[i - 1] <= w) {
                int include = values[i - 1] + dp[i - 1][w - weights[i - 1]];
                int exclude = dp[i - 1][w];
                dp[i][w] = include > exclude ? include : exclude;
            } else {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    
    return dp[n][W];
}

int knapsackOptimized(int weights[], int values[], int n, int W) {
    int dp[W + 1];
    
    for (int w = 0; w <= W; w++) dp[w] = 0;
    
    for (int i = 0; i < n; i++) {
        for (int w = W; w >= weights[i]; w--) {
            int include = values[i] + dp[w - weights[i]];
            if (include > dp[w]) dp[w] = include;
        }
    }
    
    return dp[W];
}
```

### 最长公共子序列

```c
int lcs(char *X, char *Y, int m, int n) {
    int dp[m + 1][n + 1];
    
    for (int i = 0; i <= m; i++) {
        for (int j = 0; j <= n; j++) {
            if (i == 0 || j == 0) {
                dp[i][j] = 0;
            } else if (X[i - 1] == Y[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = dp[i - 1][j] > dp[i][j - 1] ? 
                           dp[i - 1][j] : dp[i][j - 1];
            }
        }
    }
    
    return dp[m][n];
}
```

### 最长递增子序列

```c
int lis(int arr[], int n) {
    int dp[n];
    
    for (int i = 0; i < n; i++) dp[i] = 1;
    
    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (arr[j] < arr[i] && dp[j] + 1 > dp[i]) {
                dp[i] = dp[j] + 1;
            }
        }
    }
    
    int max = 0;
    for (int i = 0; i < n; i++) {
        if (dp[i] > max) max = dp[i];
    }
    
    return max;
}
```

### 编辑距离

```c
int editDistance(char *s1, char *s2, int m, int n) {
    int dp[m + 1][n + 1];
    
    for (int i = 0; i <= m; i++) dp[i][0] = i;
    for (int j = 0; j <= n; j++) dp[0][j] = j;
    
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1];
            } else {
                int insert = dp[i][j - 1];
                int delete = dp[i - 1][j];
                int replace = dp[i - 1][j - 1];
                
                int min = insert < delete ? insert : delete;
                min = min < replace ? min : replace;
                dp[i][j] = 1 + min;
            }
        }
    }
    
    return dp[m][n];
}
```

### 最大子数组和

```c
int maxSubArray(int arr[], int n) {
    int dp[n];
    dp[0] = arr[0];
    int max = dp[0];
    
    for (int i = 1; i < n; i++) {
        if (dp[i - 1] > 0) {
            dp[i] = dp[i - 1] + arr[i];
        } else {
            dp[i] = arr[i];
        }
        if (dp[i] > max) max = dp[i];
    }
    
    return max;
}

int maxSubArrayOptimized(int arr[], int n) {
    int sum = arr[0], max = arr[0];
    
    for (int i = 1; i < n; i++) {
        if (sum > 0) {
            sum += arr[i];
        } else {
            sum = arr[i];
        }
        if (sum > max) max = sum;
    }
    
    return max;
}
```

### 矩阵链乘法

```c
int matrixChain(int dims[], int n) {
    int dp[n][n];
    
    for (int i = 1; i < n; i++) dp[i][i] = 0;
    
    for (int len = 2; len < n; len++) {
        for (int i = 1; i < n - len + 1; i++) {
            int j = i + len - 1;
            dp[i][j] = INT_MAX;
            
            for (int k = i; k < j; k++) {
                int cost = dp[i][k] + dp[k + 1][j] + 
                           dims[i - 1] * dims[k] * dims[j];
                if (cost < dp[i][j]) dp[i][j] = cost;
            }
        }
    }
    
    return dp[1][n - 1];
}
```

## 记忆化搜索

```c
#define MAX 1000
int memo[MAX][MAX];

int lcsMemo(char *X, char *Y, int m, int n) {
    if (m == 0 || n == 0) return 0;
    if (memo[m][n] != -1) return memo[m][n];
    
    if (X[m - 1] == Y[n - 1]) {
        memo[m][n] = 1 + lcsMemo(X, Y, m - 1, n - 1);
    } else {
        int left = lcsMemo(X, Y, m - 1, n);
        int right = lcsMemo(X, Y, m, n - 1);
        memo[m][n] = left > right ? left : right;
    }
    
    return memo[m][n];
}
```

## 状态压缩DP

### 旅行商问题

```c
int tsp(int dist[][4], int n) {
    int dp[1 << n][n];
    
    for (int mask = 0; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            dp[mask][i] = INT_MAX;
        }
    }
    
    dp[1][0] = 0;
    
    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue;
            if (dp[mask][i] == INT_MAX) continue;
            
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;
                int newMask = mask | (1 << j);
                int newDist = dp[mask][i] + dist[i][j];
                if (newDist < dp[newMask][j]) {
                    dp[newMask][j] = newDist;
                }
            }
        }
    }
    
    int result = INT_MAX;
    int finalMask = (1 << n) - 1;
    for (int i = 1; i < n; i++) {
        if (dp[finalMask][i] != INT_MAX) {
            int total = dp[finalMask][i] + dist[i][0];
            if (total < result) result = total;
        }
    }
    
    return result;
}
```

## DP问题分类

| 类型 | 问题 | 状态定义 |
|------|------|---------|
| 线性DP | 最大子数组和、LIS | 一维状态 |
| 区间DP | 矩阵链乘法、回文分割 | 区间[i,j] |
| 树形DP | 树的最大独立集 | 树上递推 |
| 状态压缩 | TSP、棋盘覆盖 | 位运算表示状态 |
| 数位DP | 数字统计 | 按位处理 |
