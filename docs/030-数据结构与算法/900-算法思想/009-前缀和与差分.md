# 前缀和与差分

## 概述

前缀和用于快速计算区间和，差分用于快速区间修改，两者互为逆运算。

## 前缀和

### 一维前缀和

```c
void prefixSum(int arr[], int n, int prefix[]) {
    prefix[0] = arr[0];
    for (int i = 1; i < n; i++) {
        prefix[i] = prefix[i - 1] + arr[i];
    }
}

int rangeSum(int prefix[], int l, int r) {
    if (l == 0) return prefix[r];
    return prefix[r] - prefix[l - 1];
}
```

### 前缀和求区间和

```c
int* createPrefix(int arr[], int n) {
    int *prefix = (int*)malloc(sizeof(int) * (n + 1));
    prefix[0] = 0;
    
    for (int i = 0; i < n; i++) {
        prefix[i + 1] = prefix[i] + arr[i];
    }
    
    return prefix;
}

int querySum(int prefix[], int l, int r) {
    return prefix[r + 1] - prefix[l];
}
```

### 二维前缀和

```c
void prefixSum2D(int matrix[][100], int m, int n, int prefix[][101]) {
    for (int i = 0; i <= m; i++) prefix[i][0] = 0;
    for (int j = 0; j <= n; j++) prefix[0][j] = 0;
    
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            prefix[i][j] = matrix[i - 1][j - 1] 
                         + prefix[i - 1][j] 
                         + prefix[i][j - 1] 
                         - prefix[i - 1][j - 1];
        }
    }
}

int rangeSum2D(int prefix[][101], int x1, int y1, int x2, int y2) {
    return prefix[x2 + 1][y2 + 1] 
         - prefix[x1][y2 + 1] 
         - prefix[x2 + 1][y1] 
         + prefix[x1][y1];
}
```

## 差分

### 一维差分

```c
void difference(int arr[], int n, int diff[]) {
    diff[0] = arr[0];
    for (int i = 1; i < n; i++) {
        diff[i] = arr[i] - arr[i - 1];
    }
}

void recoverArray(int diff[], int n, int arr[]) {
    arr[0] = diff[0];
    for (int i = 1; i < n; i++) {
        arr[i] = arr[i - 1] + diff[i];
    }
}
```

### 区间更新

```c
void rangeUpdate(int diff[], int l, int r, int value) {
    diff[l] += value;
    if (r + 1 < 100000) {
        diff[r + 1] -= value;
    }
}

int* batchUpdate(int n, int updates[][3], int m) {
    int *diff = (int*)calloc(n + 1, sizeof(int));
    
    for (int i = 0; i < m; i++) {
        int l = updates[i][0];
        int r = updates[i][1];
        int value = updates[i][2];
        
        diff[l] += value;
        diff[r + 1] -= value;
    }
    
    int *result = (int*)malloc(sizeof(int) * n);
    result[0] = diff[0];
    for (int i = 1; i < n; i++) {
        result[i] = result[i - 1] + diff[i];
    }
    
    free(diff);
    return result;
}
```

### 二维差分

```c
void diff2D(int diff[][101], int x1, int y1, int x2, int y2, int value) {
    diff[x1][y1] += value;
    diff[x2 + 1][y1] -= value;
    diff[x1][y2 + 1] -= value;
    diff[x2 + 1][y2 + 1] += value;
}

void recover2D(int diff[][101], int m, int n, int result[][100]) {
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
            result[i - 1][j - 1] = diff[i][j];
        }
    }
}
```

## C++ 实现

```cpp
class PrefixSum {
private:
    std::vector<long long> prefix;
    
public:
    PrefixSum(const std::vector<int>& arr) {
        prefix.resize(arr.size() + 1);
        prefix[0] = 0;
        for (size_t i = 0; i < arr.size(); i++) {
            prefix[i + 1] = prefix[i] + arr[i];
        }
    }
    
    long long query(int l, int r) {
        return prefix[r + 1] - prefix[l];
    }
};

class Difference {
private:
    std::vector<int> diff;
    
public:
    Difference(int n) : diff(n + 1, 0) {}
    
    void update(int l, int r, int value) {
        diff[l] += value;
        if (r + 1 < (int)diff.size()) {
            diff[r + 1] -= value;
        }
    }
    
    std::vector<int> getResult() {
        std::vector<int> result(diff.size() - 1);
        result[0] = diff[0];
        for (size_t i = 1; i < result.size(); i++) {
            result[i] = result[i - 1] + diff[i];
        }
        return result;
    }
};
```

## 应用：和为K的子数组

```c
int subarraySum(int nums[], int n, int k) {
    int *prefix = createPrefix(nums, n);
    
    int count = 0;
    for (int i = 0; i < n; i++) {
        for (int j = i; j < n; j++) {
            if (querySum(prefix, i, j) == k) {
                count++;
            }
        }
    }
    
    free(prefix);
    return count;
}

int subarraySumOptimized(int nums[], int n, int k) {
    int count = 0;
    int sum = 0;
    
    int *hashMap = (int*)calloc(20001, sizeof(int));
    hashMap[10000] = 1;
    
    for (int i = 0; i < n; i++) {
        sum += nums[i];
        
        int target = sum - k + 10000;
        if (target >= 0 && target <= 20000) {
            count += hashMap[target];
        }
        
        hashMap[sum + 10000]++;
    }
    
    free(hashMap);
    return count;
}
```

## 应用：航班预订统计

```c
int* corpFlightBookings(int bookings[][3], int m, int n, int *returnSize) {
    int *diff = (int*)calloc(n + 2, sizeof(int));
    
    for (int i = 0; i < m; i++) {
        int first = bookings[i][0];
        int last = bookings[i][1];
        int seats = bookings[i][2];
        
        diff[first] += seats;
        diff[last + 1] -= seats;
    }
    
    int *result = (int*)malloc(sizeof(int) * n);
    *returnSize = n;
    
    result[0] = diff[1];
    for (int i = 1; i < n; i++) {
        result[i] = result[i - 1] + diff[i + 1];
    }
    
    free(diff);
    return result;
}
```

## 时间复杂度

| 操作 | 前缀和 | 差分 |
|------|--------|------|
| 预处理 | O(n) | O(n) |
| 区间查询 | O(1) | - |
| 区间更新 | - | O(1) |
| 还原数组 | - | O(n) |

## 应用场景

1. **区间求和**：快速查询
2. **区间更新**：批量修改
3. **二维矩阵**：子矩阵和
4. **差分约束**：不等式系统
